## [211 Design Add and Search Words Data Structure](https://leetcode.com/problems/design-add-and-search-words-data-structure/description/)

class TrieNode {
  #children = new Map();

  isEndOfWord = false;

  getChild(key) {
    return this.#children.get(key);
  }

  ensureChild(key) {
    let child = this.#children.get(key);

    if (!child) {
      child = new TrieNode();
      this.#children.set(key, child);
    }

    return child;
  }

  getChildren() {
    return this.#children.values().toArray();
  }
}

class WordDictionary {
  #root = new TrieNode();

  addWord(word) {
    let node = this.#root;

    for (const char of word) {
      node = node.ensureChild(char);
    }

    node.isEndOfWord = true;
  }

  search(word) {
    return this.#dfs(0, word, this.#root);
  }

  #dfs(index, word, node) {
    if (index === word.length) {
      return node.isEndOfWord;
    }

    const char = word[index];

    if (char === '.') {
      return node
        .getChildren()
        .some((child) => this.#dfs(index + 1, word, child));
    }

    const child = node.getChild(char);

    return !child ? false : this.#dfs(index + 1, word, child);
  }
}
