## [208 Implement Trie (Prefix Tree)](https://leetcode.com/problems/implement-trie-prefix-tree/description/)

class TrieNode {
  #children = new Map();

  isEndOfWord = false;

  hasChild(key) {
    return this.#children.has(key);
  }

  getChild(key) {
    return this.#children.get(key);
  }

  getOrCreateChild(char) {
    let child = this.#children.get(char);

    if (!child) {
      child = new TrieNode();
      this.#children.set(char, child);
    }

    return child;
  }
}

class Trie {
  root = new TrieNode();

  insert(word) {
    let node = this.root;

    for (const char of word) {
      node = node.getOrCreateChild(char);
    }

    node.isEndOfWord = true;
  }

  search(word) {
    return this.#findNode(word)?.isEndOfWord === true;
  }

  startsWith(prefix) {
    return this.#findNode(prefix) !== null;
  }

  #findNode(word) {
    let node = this.root;

    for (const char of word) {
      node = node.getChild(char);

      if (!node) return null;
    }

    return node;
  }
}
