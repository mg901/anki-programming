```js
// - Time: O(n)
// - Space: O(h)
function lexicalOrder(n) {
  const result = [];

  for (let i = 1; i <= 9; i += 1) {
    generate(i, n, result);
  }

  return result;
}

function generate(current, limit, result) {
  if (current > limit) return;

  result.push(current);

  for (let digit = 0; digit <= 9; digit += 1) {
    const next = current * 10 + digit;
    if (next > limit) break;

    generate(next, limit, result);
  }
}
```

## [386 Lexicographical Numbers 'space-optimized'](https://leetcode.com/problems/lexicographical-numbers/description/)

```js
// Explanation:
//  - Neetcode: https://youtu.be/QihtII-FvLo

// - Time: O(n)
// - Space: O(1)
function lexicalOrder(n) {
  const RADIX = 10;
  let current = 1;
  const sorted = [];

  for (let i = 0; i < n; i += 1) {
    sorted.push(current);

    if (current * RADIX <= n) {
      current *= RADIX;
    } else {
      while (current % RADIX === 9 || current === n) {
        current = Math.floor(current / RADIX);
      }

      current += 1;
    }
  }

  return sorted;
}
```
