## [1022 Sum of Root To Leaf Binary Numbers](https://leetcode.com/problems/sum-of-root-to-leaf-binary-numbers/description/?envType=problem-list-v2&envId=vm7q3te3)

<!-- notecardId: 1762708338510 -->

```js
// - Time: O(n)
// - Space: O(h)
function sumRootToLeaf(root) {
  return dfs(root, 0);

  function dfs(node, decimal) {
    if (!node) return 0;
    decimal = (decimal << 1) + node.val;

    if (!node.left && !node.right) {
      return decimal;
    }

    return dfs(node.left, decimal) + dfs(node.right, decimal);
  }
}
```
