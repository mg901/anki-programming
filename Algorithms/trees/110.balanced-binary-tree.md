// - Time: O(n)
// - Space: O(h)
function isBalanced(root) {
  return checkBalanced(root) !== -1;
}

function dfs(node) {
  if (!node) return 0;

  const left = dfs(node.left);
  const right = dfs(node.right);

  if (left === -1 || right === -1 || Math.abs(left - right) > 1) {
    return -1;
  }

  return 1 + Math.max(left, right);
}
