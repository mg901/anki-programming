## [74 Search a 2D Matrix](https://leetcode.com/problems/search-a-2d-matrix/description/)

<!-- notecardId: 1765664560548 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/Ber2pi2C0j0

// Approach: Virtual indexing

// - Time: O(log(m * n))
// - Space: O(1)
function searchMatrix(matrix, target) {
  const rows = matrix.length;
  const cols = matrix[0].length;
  let left = 0;
  let right = cols * rows - 1;

  while (left <= right) {
    const mid = left + ((right - left) >> 1); // Virtual Index
    const val = matrix[Math.floor(mid / cols)][mid % cols];

    if (val === target) {
      return true;
    }

    if (val < target) {
      left = mid + 1;
    } else {
      right = mid - 1;
    }
  }

  return false;
}
```
