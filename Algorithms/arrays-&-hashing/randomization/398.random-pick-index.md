## [398 Random Pick Index](https://leetcode.com/problems/random-pick-index/description/)

<!-- notecardId: 1766866349708 -->

```js
class Solution {
  // - Time: O(n)
  // - Space: O(n)
  constructor(nums) {
    this.map = new Map();

    for (let i = 0; i < nums.length; i += 1) {
      const num = nums[i];

      if (!this.map.has(num)) {
        this.map.set(num, []);
      }

      this.map.get(num).push(i);
    }
  }

  // - Time: O(1)
  pick(target) {
    const indices = this.map.get(target);
    const n = indices.length;
    const randomIdx = Math.floor(Math.random() * n);

    return indices[randomIdx];
  }
}
```
