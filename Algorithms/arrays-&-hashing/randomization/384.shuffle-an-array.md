## [384 Shuffle an Array](https://leetcode.com/problems/shuffle-an-array/description/)

<!-- notecardId: 1766866352384 -->

```js
// Algorithm: Fisher-Yates

// Explanation:
// - Bro Code: https://youtu.be/FGAUekwri1Q

// - Time: O(n)
// - Space: O(1)
class Solution {
  constructor(nums) {
    this.origin = [...nums];
    this.nums = [...nums];
  }

  reset() {
    this.nums = [...this.origin];

    return this.nums;
  }

  shuffle() {
    const n = this.nums.length;

    for (let i = 0; i < n; i += 1) {
      swap(this.nums, i, randomInt(i, n));
    }

    return this.nums;
  }
}

function randomInt(min, max) {
  return Math.floor(Math.random() * (max - min));
}

function swap(arr, i, j) {
  if (i === j) return;

  const temp = arr[j];
  arr[j] = arr[i];
  arr[i] = temp;
}
```
