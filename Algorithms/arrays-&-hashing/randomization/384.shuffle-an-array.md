## [384 Shuffle an Array](https://leetcode.com/problems/shuffle-an-array/description/)

<!-- notecardId: 1768601298385 -->

```js
// Algorithm: Fisher-Yates

// Explanation:
// - Bro Code: https://youtu.be/FGAUekwri1Q
// - Adam Khoury: https://youtu.be/tLxBwSL3lPQ

// - Time: O(n)
// - Space: O(1)
class Solution {
  constructor(nums) {
    this.original = [...nums];
    this.nums = [...nums];
  }

  reset() {
    this.nums = [...this.original];

    return this.nums;
  }

  shuffle() {
    const n = this.nums.length;

    for (let i = n - 1; i > 0; i -= 1) {
      swap(this.nums, i, randomInt(i));
    }

    return this.nums;
  }
}

function randomInt(max) {
  return Math.floor(Math.random() * (max + 1));
}

function swap(arr, i, j) {
  if (i === j) return;

  const temp = arr[j];
  arr[j] = arr[i];
  arr[i] = temp;
}
```
