## [442 Find All Duplicates in an Array](https://leetcode.com/problems/find-all-duplicates-in-an-array/description/)

<!-- notecardId: 1768864003488 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/Y8x0iAVEITo

// - Technique: Index Marking

// - Time: O(n)
// - Space: O(1)
function findDuplicates(nums) {
  const n = nums.length;
  const result = [];

  for (let i = 0; i < n; i += 1) {
    const num = Math.abs(nums[i]);
    const index = num - 1;

    if (nums[index] < 0) {
      result.push(num);
    }

    nums[index] = -Math.abs(nums[index]);
  }

  return result;
}
```
