## [26 Remove Duplicates from Sorted Array](https://leetcode.com/problems/remove-duplicates-from-sorted-array/description/)

<!-- notecardId: 1766922314071 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/DEJAZBq0FDA

// Pattern: Two Pointers

// Sub-pattern: Slow / Fast
//    Technique: Stable Partition
//        Specific: Duplicate Skipping

// - Time: O(n)
// - Space: O(1)
function removeDuplicates(nums) {
  const n = nums.length;
  if (n < 2) return nums;

  let k = 1;

  for (let i = 1; i < n; i += 1) {
    if (nums[i] !== nums[k - 1]) {
      nums[k] = nums[i];
      k += 1;
    }
  }

  return k;
}
```
