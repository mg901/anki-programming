```js
// Explanation:
// - Neetcode: https://youtu.be/BHr381Guz3Y

// Pattern: Two Pointers

// Sub-pattern: Two Ends
//    Technique: Reverse Array (Three-Reverse Trick)

// - Time: O(n)
// - Space: O(1)
function rotate(nums, k) {
  k %= nums.length;

  if (k === 0) return;

  reverse(nums);
  reverse(nums, 0, k - 1);
  reverse(nums, k);
}

function reverse(arr, left = 0, right = arr.length - 1) {
  while (left < right) {
    [arr[left], arr[right]] = [arr[right], arr[left]];

    left += 1;
    right -= 1;
  }
}
```

## [189 Rotate Array 'juggling'](https://leetcode.com/problems/rotate-array/description/)

<!-- notecardId: 1769259918748 -->

```js
// - Algorithm: Juggling

// - Time: O(n)
// - Space: O(1);
function rotate(nums, k) {
  const n = nums.length;

  k %= n;
  if (k === 0) return;

  const count = gcd(k, n);

  for (let start = 0; start < count; start += 1) {
    let current = start;
    const temp = nums[start];

    while (true) {
      // to left (current + k) % n
      // to right (current - k + n) % n
      const next = (current - k + n) % n;
      if (next === start) break;

      nums[current] = nums[next];
      current = next;
    }

    nums[current] = temp;
  }
}

function gcd(a, b) {
  while (b > 0) {
    const temp = b;
    b = a % b;
    a = temp;
  }

  return a;
}
```

## [189 Rotate Array 'splice'](https://leetcode.com/problems/rotate-array/description/)

<!-- notecardId: 1767995108429 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/BHr381Guz3Y

// - Time: O(n)
// - Space: O(n)
function rotate(nums, k) {
  const n = nums.length;

  k %= n;
  if (k === 0) return;

  const moved = nums.splice(n - k, k);
  nums.splice(0, 0, ...moved);
}
```
