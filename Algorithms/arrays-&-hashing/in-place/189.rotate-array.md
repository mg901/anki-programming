## [189 Rotate Array](https://leetcode.com/problems/rotate-array/description/)

<!-- notecardId: 1767990982839 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/BHr381Guz3Y

// Pattern: Two Pointers

// Sub-pattern: Two Ends
//    Technique: Reverse Array (Three-Reverse Trick)

// - Time: O(n)
// - Space: O(1)
function rotate(nums, k) {
  const n = nums.length;
  k %= n;

  if (k === 0) return;

  reverse(nums);
  reverse(nums, 0, k - 1);
  reverse(nums, k);
}

function reverse(arr, left = 0, right = arr.length - 1) {
  while (left < right) {
    swap(arr, left, right);

    left += 1;
    right -= 1;
  }
}

function swap(arr, i, j) {
  if (i === j) return;

  const temp = arr[j];
  arr[j] = arr[i];
  arr[i] = temp;
}
```

## [189 Rotate Array 'splice'](https://leetcode.com/problems/rotate-array/description/)

<!-- notecardId: 1767990982843 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/BHr381Guz3Y

// - Time: O(n)
// - Space: O(n)
function rotate(nums, k) {
  const n = nums.length;
  k %= n;

  if (k === 0) return;

  const moved = nums.splice(n - k, k);
  nums.splice(0, 0, ...moved);
}
```
