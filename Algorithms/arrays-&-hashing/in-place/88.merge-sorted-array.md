## [88 Merge Sorted Array](https://leetcode.com/problems/merge-sorted-array/description/)

<!-- notecardId: 1764532830056 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/P1Ic85RarKY

// Pattern: Two Pointers
//   Sub-pattern: Merge Walk

// - Time: O(n + m)
// - Space: O(1)
function merge(nums1, m, nums2, n) {
  let i = m - 1;
  let j = n - 1;
  let k = m + n - 1;

  while (j >= 0) {
    if (nums1[i] > nums2[j]) {
      nums1[k] = nums1[i];
      i -= 1;
    } else {
      nums1[k] = nums2[j];
      j -= 1;
    }

    k -= 1;
  }
}
```
