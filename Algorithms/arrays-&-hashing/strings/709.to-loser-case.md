## [709 To Lower Case](https://leetcode.com/problems/to-lower-case/description/)

<!-- notecardId: 1767347501953 -->

```js
// - Time: O(n)
// - Space: O(1)
function toLowerCase(s) {
  const capA = 'A'.codePointAt(0);
  const capZ = 'Z'.codePointAt(0);
  const lowA = 'a'.codePointAt(0);

  let result = '';

  for (const char of s) {
    const code = char.codePointAt(0);

    if (code >= capA && code <= capZ) {
      result += String.fromCodePoint(code + lowA - capA);
    } else {
      result += char;
    }
  }

  return result;
}
```
