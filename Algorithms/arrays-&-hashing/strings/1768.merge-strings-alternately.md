## [1768 Merge Strings Alternately](https://leetcode.com/problems/merge-strings-alternately/description/)

<!-- notecardId: 1765411107586 -->

```js
// - Time: O(m + n)
// - Space: O(m + n);
function mergeAlternately(word1, word2) {
  const m = word1.length;
  const n = word2.length;
  const merged = new Array(m + n - 1);

  let i = 0;
  let j = 0;

  while (i < m || j < n) {
    if (i < m) {
      merged.push(word1[i]);
      i += 1;
    }

    if (j < n) {
      merged.push(word2[j]);
      j += 1;
    }
  }

  return merged.join('');
}
```
