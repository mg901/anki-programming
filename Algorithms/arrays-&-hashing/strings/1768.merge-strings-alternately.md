## [1768 Merge Strings Alternately](https://leetcode.com/problems/merge-strings-alternately/description/)

<!-- notecardId: 1765696628081 -->

```js
// - Time: O(m + n)
// - Space: O(m + n);
function mergeAlternately(word1, word2) {
  const m = word1.length;
  const n = word2.length;
  let result = '';

  let i = 0;
  let j = 0;

  while (i < m || j < n) {
    if (i < m) {
      result += word1[i];
      i += 1;
    }

    if (j < n) {
      result += word2[j];
      j += 1;
    }
  }

  return result;
}
```
