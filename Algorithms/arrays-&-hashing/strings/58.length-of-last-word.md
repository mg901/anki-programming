## [58 Length of Last Word](https://leetcode.com/problems/length-of-last-word/description)

<!-- notecardId: 1768741416155 -->

```js
// - Time: O(n)
// - Space: O(1)
function lengthOfLastWord(s) {
  const SPACE = ' ';
  let length = 0;
  let i = s.length - 1;

  while (i >= 0 && s[i] === SPACE) {
    i -= 1;
  }

  while (i >= 0 && s[i] !== SPACE) {
    length += 1;
    i -= 1;
  }

  return length;
}
```
