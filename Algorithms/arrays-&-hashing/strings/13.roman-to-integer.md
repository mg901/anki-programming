## [13 Roman to Integer](https://leetcode.com/problems/roman-to-integer/description/)

<!-- notecardId: 1762257243756 -->

```js
const roman = new Map([
  ['I', 1],
  ['V', 5],
  ['X', 10],
  ['L', 50],
  ['C', 100],
  ['D', 500],
  ['M', 1000],
]);

// Explanation:
// - Neetcode: https://youtu.be/3jdxYj3DD98?si=O8G0KU154GW6mhth

// - Time: O(n)
// - Space: O(1)
function romanToInt(s) {
  const n = s.length;
  let result = 0;

  for (let i = 0; i < n; i += 1) {
    const char = s[i];

    if (i + 1 < n && roman.get(char) < roman.get(s[i + 1])) {
      result -= roman.get(char);
    } else {
      result += roman.get(char);
    }
  }

  return result;
}
```
