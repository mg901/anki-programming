## [3120 Count the Number of Special Characters](https://leetcode.com/problems/count-the-number-of-special-characters-i/description/)

<!-- notecardId: 1768741518757 -->

```js
// - Time: O(n)
// - Space: O(1)
function numberOfSpecialChars(word) {
  const ASCII_SIZE = 128;
  const A = 'A'.codePointAt(0);
  const z = 'z'.codePointAt(0);
  const a = 'a'.codePointAt(0);
  const caseShift = a - A;
  const seen = new Uint8Array(ASCII_SIZE);

  for (const char of word) {
    seen[char.codePointAt(0)] += 1;
  }

  let count = 0;

  for (let code = a; code <= z; code += 1) {
    if (seen[code] && seen[code - caseShift]) {
      count += 1;
    }
  }

  return count;
}
```
