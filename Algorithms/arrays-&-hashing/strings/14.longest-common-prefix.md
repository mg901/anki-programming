## [14 Longest Common Prefix 'vertical scanning'](https://leetcode.com/problems/longest-common-prefix/description/)

<!-- notecardId: 1766229005000 -->

```js
// Explanation:
// - One Come Man: https://youtu.be/H8A9twm07Vc

// - Algorithm: Vertical Scanning

// - Time: O(s)
//    where:
//      - s - the sum of the lengths of all strings

// - Space: O(1)
function longestCommonPrefix(strs) {
  const n = strs.length;
  let prefix = strs[0];

  for (let i = 1; i < n; i += 1) {
    while (strs[i].startsWith(prefix) === false) {
      prefix = prefix.slice(0, prefix.length - 1);

      if (prefix === '') return '';
    }
  }

  return prefix;
}
```

## [14 Longest Common Prefix 'horizontal scanning'](https://leetcode.com/problems/longest-common-prefix/description/)

```js
// Explanation:
// - One Come Man: https://youtu.be/H8A9twm07Vc

// - Algorithm: Horizontal Scanning

// - Time: O(s)
//    where:
//      - s - the sum of the lengths of all strings

// - Space: O(1)
function longestCommonPrefix(strs) {
  const n = strs.length;
  let prefix = strs[0];

  for (let i = 1; i < n; i += 1) {
    prefix = commonPrefix(prefix, strs[i]);

    if (prefix === '') return '';
  }

  return prefix;
}

function commonPrefix(prefix, str) {
  const m = prefix.length;
  const n = str.length;
  let i = 0;

  while ((i < m || i < n) && prefix[i] === str[i]) {
    i += 1;
  }

  return prefix.slice(0, i);
}
```
