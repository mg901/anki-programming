## [709 To Lower Case](https://leetcode.com/problems/to-lower-case/description/)

<!-- notecardId: 1767461933612 -->

```js
// - Time: O(n)
// - Space: O(1)
function toLowerCase(s) {
  const A = 'A'.codePointAt(0);
  const Z = 'Z'.codePointAt(0);
  const a = 'a'.codePointAt(0);
  const caseShift = a - A;

  let result = '';

  for (const char of s) {
    const code = char.codePointAt(0);

    if (code >= A && code <= Z) {
      result += String.fromCodePoint(code + caseShift);
    } else {
      result += char;
    }
  }

  return result;
}
```
