## [3498 Reverse Degree of a String](https://leetcode.com/problems/reverse-degree-of-a-string/description)

<!-- notecardId: 1765862641071 -->

```js
// - Time: O(n)
// - Space: O(1)
function reverseDegree(s) {
  const ALPHABET_SIZE = 26;
  const codeA = 'a'.codePointAt(0);
  let result = 0;

  for (let i = 0; i < s.length; i += 1) {
    const reversed = ALPHABET_SIZE - (s.codePointAt(i) - codeA);
    result += reversed * (i + 1);
  }

  return result;
}
```
