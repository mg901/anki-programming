## [242 Valid Anagram 'frequency array'](https://leetcode.com/problems/valid-anagram/)

<!-- notecardId: 1768741443085 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/9UtInBqnCgA

// - Time: O(n)
// - Space: O(1)
function isAnagram(s, t) {
  if (s.length !== t.length) return false;

  const ALPHABET_SIZE = 26;
  const codeA = 'a'.codePointAt(0);
  const counter = new Uint8Array(ALPHABET_SIZE);

  for (let i = 0; i < s.length; i += 1) {
    counter[s[i].codePointAt(0) - codeA] += 1;
    counter[t[i].codePointAt(0) - codeA] -= 1;
  }

  return counter.every((count) => count === 0);
}
```
