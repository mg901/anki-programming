## [347 Top K Frequent Elements](https://leetcode.com/problems/top-k-frequent-elements/description/)

<!-- notecardId: 1769360222173 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/YPTqKIgVk-k

// - Time: O(n)
// - Space: (n)
function topKFrequent(nums, k) {
  const n = nums.length;
  if (n === 1) return nums;

  const counter = new Map();

  for (const num of nums) {
    counter.set(num, (counter.get(num) ?? 0) + 1);
  }

  const buckets = Array.from({ length: n + 1 }, () => []);

  for (const [num, freq] of counter) {
    buckets[freq].push(num);
  }

  const maxFreq = buckets.length - 1;
  const kMostFreq = [];

  for (let i = maxFreq; i >= 0; i -= 1) {
    if (buckets[i].length === 0) continue;

    kMostFreq.push(...buckets[i]);
    if (kMostFreq.length === k) break;
  }

  return kMostFreq;
}
```
