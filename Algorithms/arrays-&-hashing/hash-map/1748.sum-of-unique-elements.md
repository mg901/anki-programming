## [1748 Sum of Unique Elements](https://leetcode.com/problems/sum-of-unique-elements/description/)

<!-- notecardId: 1764184733057 -->

```js
// - Time: O(n)
// - Space: O(n)
function sumOfUnique(nums) {
  const count = new Map();
  let sum = 0;

  for (const num of nums) {
    if (!count.has(num)) {
      count.set(num, 1);
      sum += num;
    } else if (count.get(num) === 1) {
      sum -= num;
      count.set(num, 2);
    }
  }

  return sum;
}
```
