## [1748 Sum of Unique Elements](https://leetcode.com/problems/sum-of-unique-elements/description/)

<!-- notecardId: 1765862860220 -->

```js
// - Time: O(n)
// - Space: O(n)
function sumOfUnique(nums) {
  const seen = new Map();
  let sum = 0;

  for (const num of nums) {
    if (!seen.has(num)) {
      seen.set(num, 1);
      sum += num;
    } else if (seen.get(num) === 1) {
      sum -= num;
      seen.set(num, 2);
    }
  }

  return sum;
}
```
