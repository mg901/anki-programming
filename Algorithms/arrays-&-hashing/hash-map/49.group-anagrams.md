## [49 Group Anagrams](https://leetcode.com/problems/group-anagrams/description/)

<!-- notecardId: 1764190571437 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/vzdNOK2oB2E

// - Time: O(n * k)
// - Space: O(n * k)
function groupAnagrams(words) {
  const map = new Map();

  for (const word of words) {
    const key = createKey(word);

    if (!map.has(key)) {
      map.set(key, []);
    }

    map.get(key).push(word);
  }

  return map.values().toArray();
}

// Use it to reduce time complexity from O(k log n) to O(k),
// - Time: O(k)
//    where: k = word.length
// - Space: O(1)
const ALPHABET_SIZE = 26;
const codeA = 'a'.charCodeAt(0);

function createKey(word) {
  const counter = new Uint8Array(ALPHABET_SIZE);

  for (const char of word) {
    counter[char.charCodeAt(0) - codeA] += 1;
  }

  return counter.toString();
}
```
