## [692 Top K Frequent Words](https://leetcode.com/problems/top-k-frequent-words/description/)

<!-- notecardId: 1767991825674 -->

```js
// - Time: O(n)
// - Space: O(n)
function topKFrequent(words, k) {
  const n = words.length;
  const wordToFreq = new Map();
  const buckets = Array.from({ length: n + 1 }, () => []);

  for (const word of words) {
    wordToFreq.set(word, (wordToFreq.get(word) ?? 0) + 1);
  }

  for (const [word, freq] of wordToFreq) {
    buckets[freq].push(word);
  }

  const kMostFrequent = [];
  const m = buckets.length;

  for (let i = m - 1; i >= 0 && k > 0; i -= 1) {
    if (!buckets[i].length) continue;

    buckets[i].sort();

    for (const word of buckets[i]) {
      kMostFrequent.push(word);
      k -= 1;

      if (k === 0) break;
    }
  }

  return kMostFrequent;
}
```
