## [383 Ransom Note](https://leetcode.com/problems/ransom-note/description)

<!-- notecardId: 1764190609762 -->

```js
// - Time: O(m + n)
//   where:
//    m = ransomNote length
//    n = magazine length
// - Space: O(1) - only 26 lowercased English letters
function canConstruct(ransomNote, magazine) {
  const ALPHABET_SIZE = 26;
  const codeA = 'a'.codePointAt(0);
  const counter = new Int32Array(ALPHABET_SIZE);

  for (const char of magazine) {
    counter[char.codePointAt(0) - codeA] += 1;
  }

  for (const char of ransomNote) {
    const index = char.codePointAt(0) - codeA;
    counter[index] -= 1;

    if (counter[index] < 0) return false;
  }

  return true;
}
```
