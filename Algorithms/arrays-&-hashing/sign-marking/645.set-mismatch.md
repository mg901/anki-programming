## [645 Set Mismatch 'sign marking'](https://leetcode.com/problems/set-mismatch/description/)

```js
// Explanation:
// - Neetcode: https://youtu.be/d-ulaeRBA64

// Technique: Sign Marking

// - Time: O(n)
// - Space: O(1)
function findErrorNums(nums) {
  const n = nums.length;
  const result = [];

  for (let i = 0; i < n; i += 1) {
    const num = Math.abs(nums[i]);
    const index = num - 1;

    if (nums[index] < 0) {
      result.push(num);
    }

    nums[index] = -Math.abs(nums[index]);
  }

  for (let i = 0; i < n; i += 1) {
    if (nums[i] > 0) {
      result.push(i + 1);
    }
  }

  return result;
}
```

## [645 Set Mismatch 'hash set'](https://leetcode.com/problems/set-mismatch/description/)

<!-- notecardId: 1769341819762 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/d-ulaeRBA64

// - Time: O(n)
// - Space: O(n)
function findErrorNums(nums) {
  const n = nums.length;
  const seen = new Set();
  const result = [];

  for (let i = 0; i < n; i += 1) {
    const num = nums[i];

    if (seen.has(num)) {
      result.push(num);
    }

    seen.add(num);
  }

  for (let i = 1; i <= n; i += 1) {
    if (!seen.has(i)) {
      result.push(i);
      break;
    }
  }

  return result;
}
```
