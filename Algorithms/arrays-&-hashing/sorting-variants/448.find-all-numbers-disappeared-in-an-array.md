## [448 Find All Numbers Disappeared in an Array 'cyclic-sort'](https://leetcode.com/problems/find-all-numbers-disappeared-in-an-array/description/)

```js
// - Neetcode: https://youtu.be/8i-f24YFWC4

// - Technique: Cyclic Sort

// - Note:
// Not optimal here. For learning Cyclic Sort only.

// - Time: O(n)
// - Space: O(1)
function findDisappearedNumbers(nums) {
  cyclicSort(nums);

  const result = [];

  for (let i = 0; i < nums.length; i += 1) {
    if (nums[i] !== i + 1) {
      result.push(i + 1);
    }
  }

  return result;
}

// - Time: O(n)
// - Space: O(1)
function cyclicSort(nums) {
  let i = 0;

  while (i < nums.length) {
    const correctIdx = nums[i] - 1;

    if (nums[i] !== nums[correctIdx]) {
      [nums[correctIdx], nums[i]] = [nums[i], nums[correctIdx]];
    } else {
      i += 1;
    }
  }
}
```
