## [561 Array Partition 'countring-sort'](https://leetcode.com/problems/array-partition/description/)

<!-- notecardId: 1764279450210 -->

```js
// - Time: O(n + k)
//   where:
//    - n = nums.length
//    - k = counter.length

// - Space: O(k)
//   where:
//    - k = counter.length;
function arrayPairSum(nums) {
  const n = nums.length;
  const min = Math.min(...nums);
  const max = Math.max(...nums);

  const length = max - min + 1;
  const counter = new Array(length).fill(0);

  for (let i = 0; i < n; i += 1) {
    counter[nums[i] - min] += 1;
  }

  let isOdd = true;
  let sum = 0;

  for (let i = 0; i < length; i += 1) {
    while (counter[i] > 0) {
      if (isOdd) sum += i + min;
      isOdd = !isOdd;
      counter[i] -= 1;
    }
  }

  return sum;
}
```
