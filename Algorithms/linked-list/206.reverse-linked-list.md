## [206 Reverse Linked List](https://leetcode.com/problems/reverse-linked-list/description/)

<!-- notecardId: 1760028021556 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/G0_I-ZF0S38

// - Time: O(n)
// - Space: O(1)
function reverseList(head) {
  if (!head) return null;

  let current = head;
  let prev = null;

  while (current) {
    const { next } = current;
    current.next = prev;
    prev = current;
    current = next;
  }

  return prev;
}
```
