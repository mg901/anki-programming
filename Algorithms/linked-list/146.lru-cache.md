## [146 LRU Cache 'doubly-linked-list'](https://leetcode.com/problems/lru-cache/description/)

<!-- notecardId: 1764104785327 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/7ABFKPK2hD4

class LRUCache {
  #capacity;

  #keyToNode = new Map();

  #head = { key: null, value: null };

  #tail = { key: null, value: null };

  constructor(capacity) {
    this.#capacity = capacity;
    this.#head.next = this.#tail;
    this.#tail.prev = this.#head;
  }

  // - Time: O(1)
  get(key) {
    const node = this.#keyToNode.get(key);
    if (!node) return -1;

    this.#delete(node);
    this.#prepend(node);

    return node.value;
  }

  // - Time: O(1)
  put(key, value) {
    if (this.#keyToNode.has(key)) {
      const node = this.#keyToNode.get(key);
      this.#delete(node);

      node.value = value;
      this.#prepend(node);

      return;
    }

    if (this.#keyToNode.size === this.#capacity) {
      const lru = this.#tail.prev;
      this.#keyToNode.delete(lru.key);
      this.#delete(lru);
    }

    const node = { key, value };
    this.#keyToNode.set(key, node);
    this.#prepend(node);
  }

  #prepend(node) {
    node.next = this.#head.next;
    node.prev = this.#head;

    this.#head.next.prev = node;
    this.#head.next = node;
  }

  #delete(node) {
    node.next.prev = node.prev;
    node.prev.next = node.next;
  }
}
```
