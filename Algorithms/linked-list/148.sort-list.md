## [148 Sort List 'space-optimized'](https://leetcode.com/problems/sort-list/description/)

```js
// Explanation:
// - Neetcode: https://youtu.be/TGveA1oFhrc
// - Mary Elaine Califf: https://youtu.be/IN_ZOU-LK08 (Bottom-up)

// - Time: O(n log(n))
// - Space: O(1)
function sortList(head) {
  if (!head || !head.next) return head;

  const dummy = new ListNode(0, head);
  let length = 0;

  while (head) {
    length += 1;
    head = head.next;
  }

  for (let step = 1; step < length; step *= 2) {
    let current = dummy.next;
    let tail = dummy;

    while (current) {
      const left = current;
      const right = split(left, step);
      current = split(right, step);
      tail = merge(left, right, tail);
    }
  }

  return dummy.next;
}

function split(current, k) {
  for (let i = 1; i < k && current; i += 1) {
    current = current.next;
  }

  if (!current) return null;

  const { next } = current;
  current.next = null;

  return next;
}

function merge(a, b, tail) {
  let current = tail;

  while (a && b) {
    if (a.val <= b.val) {
      current.next = a;
      current = a;
      a = a.next;
    } else {
      current.next = b;
      current = b;
      b = b.next;
    }
  }

  current.next = a || b;

  while (current.next) {
    current = current.next;
  }

  return current;
}
```
