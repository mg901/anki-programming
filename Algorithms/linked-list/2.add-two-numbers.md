## [2 Add Two Numbers](https://leetcode.com/problems/add-two-numbers/description/)

<!-- notecardId: 1760261325361 -->

```js
// - Time: O(n + m)
// - Space:
//  - O(1) - extra space
//  - O(max(n, m)) - output list
function addTwoNumbers(l1, l2) {
  const RADIX = 10;
  let carry = 0;
  let head = null;
  let tail = null;

  while (l1 || l2 || carry) {
    const a = l1 ? l1.val : 0;
    const b = l2 ? l2.val : 0;
    const sum = a + b + carry;
    const node = new Node(sum % RADIX);
    carry = Math.floor(sum / RADIX);

    if (!head) {
      head = node;
      tail = node;
    } else {
      tail.next = node;
      tail = node;
    }

    if (l1) l1 = l1.next;
    if (l2) l2 = l2.next;
  }

  return head;
}
```
