## [2 Add Two Numbers](https://leetcode.com/problems/add-two-numbers/description/)

```js
function addTwoNumbers(l1, l2) {
  let head = null;
  let tail = null;
  let carry = 0;
  const base = 10;

  while (l1 || l2 || carry) {
    const a = l1 ? l1.val : 0;
    const b = l2 ? l2.val : 0;
    const sum = a + b + carry;
    const node = new ListNode(sum % base);
    carry = (sum / base) ^ 0;

    if (!head) {
      head = tail = node;
    } else {
      tail.next = node;
      tail = node;
    }

    if (l1) l1 = l1.next;
    if (l2) l2 = l2.next;
  }

  return head;
}
```
