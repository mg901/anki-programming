## [2 Add Two Numbers](https://leetcode.com/problems/add-two-numbers/description/)

<!-- notecardId: 1741881553830 -->

```js
function addTwoNumbers(l1, l2) {
  let head = null;
  let tail = null;
  let carry = 0;

  while (l1 || l2 || carry) {
    const a = l1 ? l1.val : 0;
    const b = l2 ? l2.val : 0;
    const sum = a + b + carry;
    const base = 10;

    carry = (sum / base) ^ 0;
    const newNode = new ListNode(sum % base);

    if (!head) {
      head = tail = newNode;
    } else {
      tail.next = newNode;
      tail = newNode;
    }

    if (l1) l1 = l1.next;
    if (l2) l2 = l2.next;
  }

  return head;
}
```
