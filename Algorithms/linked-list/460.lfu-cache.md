## [460 LFU Cache](https://leetcode.com/problems/lfu-cache/description/)

```js
class ListNode {
  constructor(data, next = null, prev = null) {
    this.data = data;
    this.next = next;
    this.prev = prev;
  }
}

class DoublyLinkedList {
  constructor() {
    this.head = null;
    this.tail = null;
    this.size = 0;
  }

  get isEmpty() {
    return this.size === 0;
  }

  append(data) {
    const newNode = new ListNode(data);

    if (!this.head) {
      this.head = newNode;
      this.tail = newNode;
    } else {
      newNode.prev = this.tail;
      this.tail.next = newNode;
      this.tail = newNode;
    }

    this.size += 1;
  }

  deleteByRef(node) {
    if (node.prev) {
      node.prev.next = node.next;
    } else {
      this.head = node.next;
    }

    if (node.next) {
      node.next.prev = node.prev;
    } else {
      this.tail = node.prev;
    }

    node.prev = null;
    node.next = null;
    this.size -= 1;
  }

  removeHead() {
    if (!this.head) return null;

    const { head } = this;
    this.deleteByRef(head);

    return head;
  }
}

class LFUCache {
  #capacity;

  #size = 0;

  #keyNodeMap = new Map();

  #keyFrequencyMap = new Map();

  #frequencyListMap = new Map();

  #minFrequency = 1;

  constructor(capacity) {
    this.#capacity = capacity;
  }

  get(key) {
    if (!this.#keyNodeMap.has(key)) return -1;
    this.#updateFrequency(key);

    return this.#keyNodeMap.get(key).data.value;
  }

  put(key, value) {
    if (this.#capacity === 0) return;

    if (this.#keyNodeMap.has(key)) {
      const node = this.#keyNodeMap.get(key);
      node.data.value = value;

      this.#updateFrequency(key);

      return;
    }

    if (this.#size === this.#capacity) {
      this.#evictLFU();
    }

    const freq = 1;
    const list = this.#frequencyListMap.get(freq) ?? new DoublyLinkedList();
    list.append({ key, value });
    const node = list.tail;

    this.#frequencyListMap.set(freq, list);
    this.#keyNodeMap.set(key, node);
    this.#keyFrequencyMap.set(key, freq);
    this.#minFrequency = freq;
    this.#size += 1;
  }

  #updateFrequency(key) {
    const freq = this.#keyFrequencyMap.get(key);
    const node = this.#keyNodeMap.get(key);
    const list = this.#frequencyListMap.get(freq);

    list.deleteByRef(node);

    if (list.isEmpty) {
      this.#frequencyListMap.delete(freq);

      if (this.#minFrequency === freq) {
        this.#minFrequency += 1;
      }
    }

    const newFreq = freq + 1;
    const newList =
      this.#frequencyListMap.get(newFreq) ?? new DoublyLinkedList();
    newList.append(node.data);
    const newNode = newList.tail;

    this.#frequencyListMap.set(newFreq, newList);
    this.#keyNodeMap.set(key, newNode);
    this.#keyFrequencyMap.set(key, newFreq);
  }

  #evictLFU() {
    const list = this.#frequencyListMap.get(this.#minFrequency);
    const head = list.removeHead();

    const { key } = head.data;
    this.#keyNodeMap.delete(key);
    this.#keyFrequencyMap.delete(key);

    if (list.isEmpty) {
      this.#frequencyListMap.delete(this.#minFrequency);
    }

    this.#size -= 1;
  }
}
```
