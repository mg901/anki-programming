## [707 Design Linked List 'sll'](https://leetcode.com/problems/design-linked-list/description/)

```js
// Explanation:
// - Neetcode: https://youtu.be/Wf4QhpdVFQo

class ListNode {
  constructor(val, next = null) {
    this.val = val;
    this.next = next;
  }
}

class MyLinkedList {
  head = null;

  tail = null;

  length = 0;

  get(index) {
    return this.#findAtIndex(index)?.val ?? -1;
  }

  addAtHead(val) {
    const newNode = new ListNode(val);

    if (!this.head) {
      this.tail = newNode;
    }

    newNode.next = this.head;
    this.head = newNode;
    this.length += 1;
  }

  addAtTail(val) {
    const newNode = new ListNode(val);

    if (!this.head) {
      this.head = newNode;
    } else {
      this.tail.next = newNode;
    }

    this.tail = newNode;
    this.length += 1;
  }

  addAtIndex(index, val) {
    if (index < 0 || index > this.length) return;

    if (index === 0) {
      this.addAtHead(val);

      return;
    }

    if (index === this.length) {
      this.addAtTail(val);

      return;
    }

    const prevNode = this.#findAtIndex(index - 1);
    prevNode.next = new ListNode(val, prevNode.next);
    this.length += 1;
  }

  deleteAtIndex(index) {
    if (index < 0 || index >= this.length) return;

    if (index === 0) {
      this.head = this.head.next;

      if (!this.head) this.tail = null;
      this.length -= 1;

      return;
    }

    const prevNode = this.#findAtIndex(index - 1);
    const toDelete = prevNode.next;
    prevNode.next = toDelete.next;

    if (!prevNode.next) {
      this.tail = prevNode;
    }

    toDelete.next = null;
    this.length -= 1;
  }

  #findAtIndex(index) {
    if (index < 0 || index >= this.length) {
      return null;
    }

    let current = this.head;
    for (let i = 0; i < index; i += 1) {
      current = current.next;
    }

    return current;
  }
}
```

## [707 Design Linked List 'dll'](https://leetcode.com/problems/design-linked-list/description/)

```js
// Explanation:
// - Neetcode: https://youtu.be/Wf4QhpdVFQo

class ListNode {
  constructor(val, next = null, prev = null) {
    this.val = val;
    this.next = next;
    this.prev = prev;
  }
}

class MyLinkedList {
  head = new ListNode(null);

  tail = new ListNode(null);

  length = 0;

  constructor() {
    this.head.next = this.tail;
    this.tail.prev = this.head;
  }

  get(index) {
    return this.#findByIndex(index)?.val ?? -1;
  }

  addAtHead(val) {
    const node = new ListNode(val, this.head.next, this.head);

    this.head.next.prev = node;
    this.head.next = node;

    this.length += 1;
  }

  addAtTail(val) {
    const node = new ListNode(val, this.tail, this.tail.prev);

    this.tail.prev.next = node;
    this.tail.prev = node;

    this.length += 1;
  }

  addAtIndex(index, val) {
    if (index < 0 || index > this.length) return;

    if (index === 0) {
      this.addAtHead(val);

      return;
    }

    if (index === this.length) {
      this.addAtTail(val);

      return;
    }

    const prevNode = this.#findByIndex(index - 1);
    const newNode = new ListNode(val, prevNode.next, prevNode);

    prevNode.next.prev = newNode;
    prevNode.next = newNode;

    this.length += 1;
  }

  deleteAtIndex(index) {
    const deleted = this.#findByIndex(index);
    if (!deleted) return;

    deleted.prev.next = deleted.next;
    deleted.next.prev = deleted.prev;

    deleted.next = null;
    deleted.prev = null;

    this.length -= 1;
  }

  #findByIndex(index) {
    if (index < 0 || index >= this.length) {
      return null;
    }

    return index < this.length / 2
      ? this.#searchAtStart(index)
      : this.#searchAtEnd(index);
  }

  #searchAtStart(index) {
    let current = this.head.next;

    for (let i = 0; i < index; i += 1) {
      current = current.next;
    }

    return current;
  }

  #searchAtEnd(index) {
    let current = this.tail.prev;

    for (let i = this.length - 1; i > index; i -= 1) {
      current = current.prev;
    }

    return current;
  }
}
```
