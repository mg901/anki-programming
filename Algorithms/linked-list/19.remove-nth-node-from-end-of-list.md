## [19 Remove Nth Node From End of List](https://leetcode.com/problems/remove-nth-node-from-end-of-list/description/)

<!-- notecardId: 1764339367172 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/XVuQxVej6y8

// - Time: O(n)
// - Space: O(1)
function removeNthFromEnd(head, n) {
  if (!head) return null;

  let slow = head;
  let fast = head;

  for (let i = 0; i < n; i += 1) {
    fast = fast.next;

    if (!fast) return slow.next;
  }

  while (fast.next) {
    slow = slow.next;
    fast = fast.next;
  }

  slow.next = slow.next.next;

  return head;
}
```
