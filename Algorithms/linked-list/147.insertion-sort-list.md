## [147 Insertion Sort List](https://leetcode.com/problems/insertion-sort-list/description/)

<!-- notecardId: 1765985037626 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/Kk6mXAzqX3Y

// - Time: O(n)
// - Space: O(1)
function insertionSortList(head) {
  if (!head || !head.next) return head;

  const dummy = new ListNode(0);
  let current = head;

  while (current) {
    let sorted = dummy;
    const { next } = current;

    while (sorted?.next?.val < current.val) {
      sorted = sorted.next;
    }

    current.next = sorted.next;
    sorted.next = current;
    current = next;
  }

  return dummy.next;
}
```
