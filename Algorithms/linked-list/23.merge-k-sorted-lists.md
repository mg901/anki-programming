## [23 Merge k Sorted Lists](https://leetcode.com/problems/merge-k-sorted-lists/description/)

<!-- notecardId: 1765664773322 -->

```js
// - Time: O(n log(k))
//    where:
//      k = total number of lists
//      n =  total number of nodes across k lists
// - Space: O(k)
function mergeKLists(lists) {
  if (!lists.length) return null;

  while (lists.length > 1) {
    const merged = [];

    for (let i = 0; i < lists.length; i += 2) {
      const a = lists[i] ?? null;
      const b = lists[i + 1] ?? null;

      merged.push(merge(a, b));
    }

    lists = merged;
  }

  return lists[0];
}

function merge(a, b) {
  if (!a) return b;
  if (!b) return a;

  const dummy = new ListNode(0);
  let current = dummy;

  while (a && b) {
    if (a.val <= b.val) {
      current.next = a;
      a = a.next;
    } else {
      current.next = b;
      b = b.next;
    }

    current = current.next;
  }

  current.next = a || b;

  return dummy.next;
}
```
