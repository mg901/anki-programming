## [141 Linked List Cycle](https://leetcode.com/problems/linked-list-cycle/description/)

<!-- notecardId: 1741803808429 -->

```js
function hasCycle(head) {
  if (!head || !head.next) return false;

  let slow = head;
  let fast = head;

  while (fast?.next) {
    slow = slow.next;
    fast = fast.next.next;

    if (slow === fast) return true;
  }

  return false;
}
```
