## [9 Palindrome Number 'reversed-and-compare'](https://leetcode.com/problems/palindrome-number/description/)

<!-- notecardId: 1767194571322 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/yubRKwixN-U

// - Time: O(log(n))
// - Space: O(1)
function isPalindrome(x) {
  const RADIX = 10;

  // less than 0 or the number ends in zero (10, 20, 30 ...etc.)
  if (x < 0 || (x !== 0 && x % RADIX === 0)) return false;

  let reversed = 0;

  while (x > reversed) {
    reversed = reversed * RADIX + (x % RADIX);
    x = Math.floor(x / RADIX);
  }

  return x === reversed || x === Math.floor(reversed / RADIX);
}
```

## [9 Palindrome Number 'two-pointer numeric'](https://leetcode.com/problems/palindrome-number/description/)

```js
// Explanation:
// - Neetcode: https://youtu.be/yubRKwixN-U

// Approach: Two-pointer Numeric

// - Time: O(n)
// - Space: O(1)
function isPalindrome(x) {
  const RADIX = 10;
  if (x < 0 || (x !== 0 && x % RADIX === 0)) return false;

  /*
   *  In strictly typed languages, divisor * 10 can overflow before the comparison.
   *
   * If divisor is near the max integer, multiplying by 10 wraps to an invalid value,
   *  which can cause an infinite loop or a crash.
   */
  let divisor = 1;
  while (x / divisor >= 10) {
    divisor *= 10;
  }

  while (x > 0) {
    let left = Math.floor(x / divisor);
    let right = x % RADIX;

    if (left !== right) return false;

    x = Math.floor((x % divisor) / RADIX);
    divisor /= 100;
  }

  return true;
}
```
