## [66 Plus One](https://leetcode.com/problems/plus-one/description/)

<!-- notecardId: 1764200720056 -->

```js
// - Time: O(n)
// - Space: O(1)
function plusOne(digits) {
  const n = digits.length;

  for (let i = n - 1; i >= 0; i -= 1) {
    digits[i] += 1;

    if (digits[i] < 10) return digits;
    digits[i] = 0;
  }

  digits.unshift(1);

  return digits;
}
```
