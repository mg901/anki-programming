## [50 Pow(x, n)](https://leetcode.com/problems/powx-n/description/)

<!-- notecardId: 1762026670272 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/g9YQyYi4IQQ
// - Math Theory: https://www.notion.so/mg901/Math-359f81039b364111add11afd995ec153?p=157bbc1e073a80289e9ce86ed380af25&pm=s

// - Time: O(log n)
// - Space: O(1)
function myPow(x, n) {
  if (n === 0) return 1;

  if (n < 1) {
    x = 1 / x;
    n = -n;
  }

  let result = 1;

  while (n > 0) {
    if (n % 2 === 1) result *= x;
    x *= x;
    n = Math.floor(n / 2);
  }

  return result;
}
```
