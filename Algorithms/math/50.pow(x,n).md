## [50 Pow(x, n)](https://leetcode.com/problems/powx-n/description/)

<!-- notecardId: 1747745782328 -->

```js
function myPow(x, n) {
  if (n === 0) return 1;

  if (n < 1) {
    x = 1 / x;
    n = -n;
  }

  let result = 1;

  while (n > 0) {
    if (n % 2 === 1) result *= x;
    x *= x;
    n = Math.floor(n / 2);
  }

  return result;
}
```
