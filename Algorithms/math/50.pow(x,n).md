## [50 Pow(x, n)](https://leetcode.com/problems/powx-n/description/)

<!-- notecardId: 1762893275967 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/g9YQyYi4IQQ

// - Time: O(log n)
// - Space: O(1)
function myPow(x, n) {
  if (n === 0) return 1;

  if (n < 0) {
    x = 1 / x;
    n = -n;
  }

  let result = 1;

  while (n > 0) {
    if (n % 2 === 1) result *= x;
    x *= x;
    n = Math.floor(n / 2);
  }

  return result;
}
```
