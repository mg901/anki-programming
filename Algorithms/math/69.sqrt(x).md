## [69 Sqrt(x)](https://leetcode.com/problems/sqrtx/description/)

<!-- notecardId: 1770127525111 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/zdMhGxRWutQ

// - Time: O(log(n))
// - Space: O(1)
function mySqrt(x) {
  if (x < 2) return x;

  let left = 1;
  let right = x >> 1;

  while (left <= right) {
    const mid = left + ((right - left) >> 1);

    // Avoid `mid * mid`
    // to prevent overflow
    if (mid <= Math.floor(x / mid)) {
      left = mid + 1;
    } else {
      right = mid - 1;
    }
  }

  return right;
}
```
