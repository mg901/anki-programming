## [48 Rotate Image](https://leetcode.com/problems/rotate-image/description/)

<!-- notecardId: 1747594627210 -->

```js
function rotate(matrix) {
  const n = matrix.length;

  for (let r = 0; r < n; r += 1) {
    for (let c = r + 1; c < n; c += 1) {
      const temp = matrix[r][c];
      matrix[r][c] = matrix[c][r];
      matrix[c][r] = temp;
    }
  }

  for (let i = 0; i < n; i += 1) {
    matrix[i].reverse();
  }
}
```
