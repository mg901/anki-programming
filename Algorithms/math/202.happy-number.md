## [202 Happy Number](https://leetcode.com/problems/happy-number/description/)

```js
function isHappy(n) {
  let slow = n;
  let fast = sumOfSquares(n);

  while (fast !== 1 && slow !== fast) {
    slow = sumOfSquares(slow);
    fast = sumOfSquares(sumOfSquares(fast));
  }

  return fast === 1;
}

function sumOfSquares(num) {
  let sum = 0;

  while (num > 0) {
    let digit = num % 10;
    sum += digit ** 2;
    num = (num / 10) ^ 0;
  }

  return sum;
}
```
