## [3280 Convert Date to Binary](https://leetcode.com/problems/convert-date-to-binary/description/)

<!-- notecardId: 1764843789692 -->

```js
// - Time: O(1)
// - Space: O(1)
function convertDateToBinary(date) {
  return date.split('-').map((val) => toBinary(Number(val)).join('-'));
}

// - Time: O(n)
// - Space: O(1)
function toBinary(decimal) {
  const RADIX = 2;
  let power = 1;
  let binary = 0;

  while (decimal > 0) {
    const remainder = decimal % RADIX;
    binary += remainder * power;
    decimal = Math.trunc(decimal / RADIX);
    power *= 10;
  }

  return binary;
}
```
