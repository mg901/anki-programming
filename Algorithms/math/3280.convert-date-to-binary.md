## [3280 Convert Date to Binary](https://leetcode.com/problems/convert-date-to-binary/description/)

<!-- notecardId: 1767795797885 -->

```js
// - Time: O(1)
// - Space: O(1)
function convertDateToBinary(date) {
  const [y, m, d] = date.split('-');

  // prettier-ignore
  return [
    toBinary(Number(y)),
    toBinary(Number(m)),
    toBinary(Number(d)),
  ].join('-');
}

// - Time: O(n)
// - Space: O(1)
function toBinary(decimal) {
  let binary = 0;
  let place = 1;

  while (decimal > 0) {
    binary += (decimal & 1) * place;
    decimal >>= 1;
    place *= 10;
  }

  return binary;
}
```
