## [7 Reverse Integer](https://leetcode.com/problems/reverse-integer/description/)

<!-- notecardId: 1763913182691 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/HAgLH58IgJQ

// - Time: O(n)
// - Space: O(1)
function reverse(x) {
  const min = -(2 ** 31);
  const max = 2 ** 31 - 1;
  const RADIX = 10;
  let reversed = 0;

  while (x !== 0) {
    const remainder = x % RADIX;
    const next = reversed * RADIX + remainder;

    if (next < min || next > max) {
      return 0;
    }

    reversed = next;
    x = Math.trunc(x / RADIX);
  }

  return reversed;
}
```
