## [7 Reverse Integer](https://leetcode.com/problems/reverse-integer/description/)

<!-- notecardId: 1762462038592 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/HAgLH58IgJQ

// - Time: O(n)
// - Space: O(1)
function reverse(x) {
  const RADIX = 10;
  const min = -(2 ** 31);
  const max = 2 ** 31 - 1;
  let reversed = 0;

  while (x !== 0) {
    const remainder = x % RADIX;

    if (reversed < min / RADIX || (reversed === min && remainder < -8)) {
      return 0;
    }

    if (reversed > max / RADIX || (reversed === max && remainder > 7)) {
      return 0;
    }

    reversed = reversed * RADIX + remainder;
    x = Math.trunc(x / RADIX);
  }

  return reversed;
}
```
