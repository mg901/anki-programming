## [204 Count Primes](https://leetcode.com/problems/count-primes/description/)

<!-- notecardId: 1767608198425 -->

```js
// - Time: O(n log log n)
// - Space: O(n)
function countPrimes(n) {
  let count = 0;
  const sieve = new Array(n).fill(false, 0, 2).fill(true, 2);

  for (let i = 2; i * i < n; i += 1) {
    if (sieve[i] === false) continue;

    for (let p = i * i; p < n; p += i) {
      sieve[p] = false;
    }
  }

  for (let i = 2; i < n; i += 1) {
    if (sieve[i]) count += 1;
  }

  return count;
}
```
