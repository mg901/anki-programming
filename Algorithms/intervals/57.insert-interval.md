## [57 Insert Interval 'Linear'](https://leetcode.com/problems/insert-interval/description/)

<!-- notecardId: 1768312602589 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/A8NUOmlwOlM

// - Time: O(n)
// - Space: O(1)
function insert(intervals, newInterval) {
  const n = intervals.length;
  let [newStart, newEnd] = newInterval;

  const result = [];
  let i = 0;

  while (i < n && intervals[i][1] < newStart) {
    result.push(intervals[i]);
    i += 1;
  }

  while (i < n && intervals[i][0] <= newEnd) {
    newStart = Math.min(newStart, intervals[i][0]);
    newEnd = Math.max(newEnd, intervals[i][1]);

    i += 1;
  }
  result.push([newStart, newEnd]);

  while (i < n) {
    result.push(intervals[i]);
    i += 1;
  }

  return result;
}
```

## [57 Insert Interval 'Greedy'](https://leetcode.com/problems/insert-interval/description/)

<!-- notecardId: 1768313394245 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/A8NUOmlwOlM

// - Time: O(n)
// - Space: O(1)
function insert(intervals, newInterval) {
  let current = [...newInterval];
  const result = [];

  for (const [start, end] of intervals) {
    if (current === null || end < current[0]) {
      result.push([start, end]);
    } else if (start > current[1]) {
      result.push(current);
      result.push([start, end]);
      current = null;
    } else {
      current[0] = Math.min(current[0], start);
      current[1] = Math.max(current[1], end);
    }
  }

  // Intervals are empty or newInterval was fully merged
  // with existing intervals
  if (current !== null) {
    result.push(current);
  }

  return result;
}
```
