## [435 Non-overlapping Intervals 'Sort-by-Start'](https://leetcode.com/problems/non-overlapping-intervals/description/)

<!-- notecardId: 1767882619497 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/nONCGxWoUfM

// - Time: O(n)
// - Space: O(1)
function eraseOverlapIntervals(intervals) {
  intervals.sort((a, b) => a[0] - b[0]);

  const n = intervals.length;
  let prevEnd = intervals[0][1];
  let count = 0;

  for (let i = 1; i < n; i += 1) {
    const [start, end] = intervals[i];

    if (prevEnd > start) {
      count += 1;
      prevEnd = Math.min(prevEnd, end);
    } else {
      prevEnd = end;
    }
  }

  return count;
}
```
