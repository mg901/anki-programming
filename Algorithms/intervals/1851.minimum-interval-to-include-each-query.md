## [1851 Minimum Interval to Include Each Query](https://leetcode.com/problems/minimum-interval-to-include-each-query/description/)

<!-- notecardId: 1768298711997 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/5hQ5WWW5awQ

// - Time: O(n log(n) + m log(m))
// - Space: O(n + m)
function minInterval(intervals, queries) {
  intervals.sort((a, b) => a[0] - b[0]); // O(n log(n))

  const n = intervals.length;
  const sortedQueries = queries.toSorted((a, b) => a - b); // O(m log(m))
  const minHeap = new MinHeap((node) => node.size);
  const queryToSize = new Map();

  let i = 0;

  // O(m)
  for (const query of sortedQueries) {
    while (i < n && intervals[i][0] <= query) {
      const [start, end] = intervals[i];

      // O(log(n))
      minHeap.push({
        size: end - start + 1,
        end,
      });

      i += 1;
    }

    // O(log(n))
    while (!minHeap.isEmpty() && minHeap.top().end < query) {
      minHeap.pop();
    }

    // O(1)
    queryToSize.set(query, minHeap.isEmpty() ? -1 : minHeap.top().size);
  }

  // O(m)
  return queries.map((query) => queryToSize.get(query));
}
```
