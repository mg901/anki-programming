function minMeetingRooms(intervals) {
  const events = [];

  for (const interval of intervals) {
    events.push([interval.start, 1]);
    events.push([interval.end, -1]);
  }

  events.sort((a, b) => (a[0] === b[0] ? a[1] - b[1] : a[0] - b[0]));
