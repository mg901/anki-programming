## [508 Wiggle Sort]()

<!-- notecardId: 1740395068958 -->

```js
// Гарантирует, что результат будет отсортирован и
// будет иметь корректный порядок (a < b, c > d)
function wiggleSort(arr) {
  arr.sort((a, b) => a - b);

  for (let i = 1; i < arr.length - 1; i += 2) {
    swap(arr, i, i + 1);
  }

  return arr;
}

// or
// Гарантирует, корректный порядок (a < b, c > d)
function wiggleSort(arr) {
  for (let i = 0; i < arr.length - 1; i += 1) {
    if (i % 2 === 1 && arr[i] < arr[i + 1]) {
      swap(arr, i, i + 1);
    }
  }
}

function swap(arr, i, j) {
  const temp = arr[j];
  arr[j] = arr[i];
  arr[i] = temp;
}
```
