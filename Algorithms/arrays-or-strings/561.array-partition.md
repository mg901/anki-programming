## [561 Array Partition 'countring-sort'](https://leetcode.com/problems/array-partition/description/)

<!-- notecardId: 1763575224570 -->

```js
// - Time: O(n + k)
//   where:
//    - n = nums.length
//    - k = freq.length

// - Space: O(n + k)
//   where:
//    - n = sorted.length
//    - k = freq.length;
function arrayPairSum(nums) {
  let min = nums[0];
  let max = nums[0];

  for (const num of nums) {
    min = Math.min(min, num);
    max = Math.max(max, num);
  }

  const n = max - min + 1;
  const freq = new Array(n).fill(0);

  for (const num of nums) {
    freq[num - min] += 1;
  }

  const sorted = [];

  for (let i = 0; i < n; i += 1) {
    while (freq[i] > 0) {
      sorted.push(i + min);
      freq[i] -= 1;
    }
  }

  let sum = 0;
  for (let i = 0; i < sorted.length; i += 2) {
    sum += sorted[i];
  }

  return sum;
}
```
