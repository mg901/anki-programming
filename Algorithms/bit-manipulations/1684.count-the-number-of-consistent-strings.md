pa## [1684 Count the Number of Consistent Strings](https://leetcode.com/problems/count-the-number-of-consistent-strings/description/)

<!-- notecardId: 1767525289305 -->

```js
// - Time: O(n * m)
// - Space: O(1)
function countConsistentStrings(allowed, words) {
  const codeA = 'a'.codePointAt(0);
  let mask = 0;

  for (const char of allowed) {
    mask |= 1 << (char.codePointAt(0) - codeA);
  }

  let count = words.length;

  for (const word of words) {
    for (const char of word) {
      const shift = char.codePointAt(0) - codeA;

      if ((mask & (1 << shift)) === 0) {
        count -= 1;
        break;
      }
    }
  }

  return count;
}
```
