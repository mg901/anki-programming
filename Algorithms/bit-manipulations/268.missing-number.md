## [268 Missing Number 'brute-force'](https://leetcode.com/problems/missing-number/description/)

<!-- notecardId: 1763721263301 -->

```js
function missingNumber(nums) {
  const n = nums.length;

  for (let i = 0; i <= n; i += 1) {
    let found = false;

    for (let j = 0; j < n; j += 1) {
      if (nums[j] === i) {
        found = true;
        break;
      }
    }

    if (!found) {
      return i;
    }
  }

  return -1;
}
```

## [268 Missing Number 'xor'](https://leetcode.com/problems/missing-number/description/)

<!-- notecardId: 1762779377050 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/WnPLSRLSANE

// - Time: O(n)
// - Space: O(1)
function missingNumber(nums) {
  const n = nums.length;
  let xor = n;

  for (let i = 0; i < n; i += 1) {
    xor ^= nums[i] ^ i;
  }

  return xor;
}
```

## [268 Missing Number 'arithmetic-sequence'](https://leetcode.com/problems/missing-number/description/)

```js
// Explanation:
// - Neetcode: https://youtu.be/WnPLSRLSANE

// - Time: O(n)
// - Space: O(1)
function missingNumber(nums) {
  const n = nums.length;
  const total = (n * (n + 1)) / 2; // (n + 1 * (n + 0)) / 2
  const sum = nums.reduce((acc, num) => acc + num, 0);

  return total - sum;
}
```

## [268 Missing Number 'binary-search'](https://leetcode.com/problems/missing-number/description/)

<!-- notecardId: 1763671526748 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/WnPLSRLSANE

// - Time: O(n log n)
// - Space: O(1)
function missingNumber(nums) {
  nums.sort((a, b) => a - b);

  let left = 0;
  let right = nums.length;

  while (left < right) {
    const mid = left + ((right - left) >> 1);

    if (nums[mid] === mid) {
      left = mid + 1;
    } else {
      right = mid;
    }
  }

  return left;
}
```
