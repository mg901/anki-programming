## [67 Add Binary](https://leetcode.com/problems/add-binary/description/)

<!-- notecardId: 1762367571012 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/keuWJ47xG8g

// Time: O(max(n,m))
// Space: O(max(m,n))
function addBinary(a, b) {
  let i = a.length - 1;
  let j = b.length - 1;
  let carry = 0;
  const codeZero = '0'.codePointAt(0);
  const result = [];

  while (i >= 0 || j >= 0 || carry) {
    let sum = carry;

    if (i >= 0) {
      sum += a.codePointAt(i) - codeZero;
      i -= 1;
    }

    if (j >= 0) {
      sum += b.codePointAt(j) - codeZero;
      j -= 1;
    }

    result.push(sum & 1);
    carry = sum >> 1;
  }

  return result.reverse().join('');
}
```

## [67 Add Binary 'bigInt'](https://leetcode.com/problems/add-binary/description/)

<!-- notecardId: 1761835669832 -->

```js
// Time: O(max(n, m))
//  where:
//    n = a.length
//    m = b.length

// Space: O(max(n, m))
function addBinary(a, b) {
  return (BigInt(`0b${a}`) + BigInt(`0b${b}`)).toString(2);
}
```
