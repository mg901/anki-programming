## [238 Product of Array Except Self](https://leetcode.com/problems/product-of-array-except-self/description/)

<!-- notecardId: 1740067078960 -->

```js
const productExceptSelf = function (nums) {
  const results = [];

  let prefix = 1;
  let suffix = 1;
  const n = nums.length;

  for (let i = 0; i < n; i += 1) {
    results[i] = prefix;
    prefix *= nums[i];
  }

  for (let i = n - 1; i >= 0; i -= 1) {
    results[i] *= suffix;
    suffix *= nums[i];
  }

  return results;
};
```
