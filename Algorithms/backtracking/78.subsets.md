## [78 Subsets](https://leetcode.com/problems/subsets/description/)

<!-- notecardId: 1744910948106 -->

```js
function subsets(arr) {
  const result = [];
  backtrack(0, []);

  return result;

  function backtrack(start, subset) {
    result.push([...subset]);

    for (let i = start; i < arr.length; i += 1) {
      subset.push(arr[i]);
      backtrack(i + 1, subset);

      subset.pop();
    }
  }
}
```
