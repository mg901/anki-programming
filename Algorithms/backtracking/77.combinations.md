## [77 Combinations](https://leetcode.com/problems/combinations/description)

<!-- notecardId: 1757983479464 -->

```js
// - Time: O(C(n, k) * k)
// - Space: O(C(n, k) * k)
function combine(n, k) {
  const result = [];
  backtrack(1, []);

  return result;

  function backtrack(start, path) {
    if (path.length === k) {
      result.push([...path]);

      return;
    }

    for (let num = start; num <= n; num += 1) {
      path.push(num);
      backtrack(num + 1, path);
      path.pop();
    }
  }
}
```
