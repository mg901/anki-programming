## [Left and Right Sum Differences](https://leetcode.com/problems/left-and-right-sum-differences/description/)

<!-- notecardId: 1764860015561 -->

```js
// - Time: O(n)
// - Space: O(1)
function leftRightDifference(nums) {
  const n = nums.length;
  const total = nums.reduce((a, b) => a + b, 0);
  const result = [];
  let leftSum = 0;

  for (let i = 0; i < n; i += 1) {
    const right = total - nums[i] - leftSum;
    result.push(Math.abs(leftSum - right));
    leftSum += nums[i];
  }

  return result;
}
```
