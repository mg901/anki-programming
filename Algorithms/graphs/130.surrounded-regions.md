## [130 Surrounded Regions](https://leetcode.com/problems/surrounded-regions/description/)

<!-- notecardId: 1755513894477 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/9z2BunfoZ5Y

// Complexity:
// - Time: O(rows * cols)
// - Space: O(rows * cols)
function solve(board) {
  const rows = board.length;
  const cols = board[0].length;

  for (let r = 0; r < rows; r += 1) {
    capture(r, 0);
    capture(r, cols - 1);
  }

  for (let c = 0; c < cols; c += 1) {
    capture(0, c);
    capture(rows - 1, c);
  }

  for (let r = 0; r < rows; r += 1) {
    for (let c = 0; c < cols; c += 1) {
      if (board[r][c] === 'O') {
        board[r][c] = 'X';
      } else if (board[r][c] === 'T') {
        board[r][c] = 'O';
      }
    }
  }

  function capture(row, col) {
    if (
      row < 0 ||
      col < 0 ||
      row >= rows ||
      col >= cols ||
      board[row][col] !== 'O'
    ) {
      return;
    }

    board[row][col] = 'T';

    capture(row + 1, col);
    capture(row - 1, col);
    capture(row, col + 1);
    capture(row, col - 1);
  }
}
```
