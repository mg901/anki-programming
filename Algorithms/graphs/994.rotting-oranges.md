## [994 Rotting Oranges](https://leetcode.com/problems/rotting-oranges/description/)

<!-- notecardId: 1755288751071 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/y704fEOx0s0

// Complexity:
// - Time: O(rows * cols)
// - Space: O(rows * cols)
function orangesRotting(grid) {
  const rows = grid.length;
  const cols = grid[0].length;
  const queue = new Queue();

  let minutes = 0;
  let freshFruitsCount = 0;

  for (let r = 0; r < rows; r += 1) {
    for (let c = 0; c < cols; c += 1) {
      const value = grid[r][c];

      if (value === 2) {
        queue.enqueue([r, c]);
      } else if (value === 1) {
        freshFruitsCount += 1;
      }
    }
  }

  if (freshFruitsCount === 0) return 0;

  const directions = [
    [-1, 0],
    [0, 1],
    [1, 0],
    [0, -1],
  ];

  while (queue.size() && freshFruitsCount > 0) {
    const size = queue.size();

    for (let i = 0; i < size; i += 1) {
      const [prevRow, prevCol] = queue.dequeue();

      for (const [dr, dc] of directions) {
        const row = prevRow + dr;
        const col = prevCol + dc;

        if (
          row < 0 ||
          col < 0 ||
          row === rows ||
          col === cols ||
          grid[row][col] !== 1
        ) {
          continue;
        }

        grid[row][col] = 2;
        freshFruitsCount -= 1;
        queue.enqueue([row, col]);
      }
    }

    minutes += 1;
  }

  return freshFruitsCount !== 0 ? -1 : minutes;
}
```
