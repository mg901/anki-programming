## [261 Graph Valid Tree 'dfs'](https://neetcode.io/problems/valid-tree?list=neetcode150)

<!-- notecardId: 1768065142991 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/bXsUuownnoQ

// Complexity:
// - Time: O(V + E)
// - Space: O(V + E)
class Solution {
  validTree(n, edges) {
    if (edges.length !== n - 1) return false;

    const visited = new Set();
    const graph = createGraph(n, edges);

    if (!dfs(0, -1)) return false;

    return visited.size === n;

    function dfs(u, parent) {
      if (visited.has(u)) return false;

      visited.add(u);

      for (const v of graph[u]) {
        if (v === parent) continue;
        if (!dfs(v, u)) return false;
      }

      return true;
    }
  }
}

function createGraph(n, edges) {
  const graph = Array.from({ length: n }, () => []);

  for (const [u, v] of edges) {
    graph[u].push(v);
    graph[v].push(u);
  }

  return graph;
}
```

## [261 Graph Valid Tree 'disjoint-set'](https://neetcode.io/problems/valid-tree?list=neetcode150)

<!-- notecardId: 1766092645042 -->

```js
// Explanation:
// - Problem:
//    Neetcode: https://youtu.be/bXsUuownnoQ

// - Algorithm
//    Union Find: https://youtu.be/92UpvDXc8fs
//    Path Compression: https://youtu.be/92UpvDXc8fs?si=6xFo-1PxUNhcrIrz&t=655
class DisjointSet {
  constructor(n) {
    this.parent = Array.from({ length: n }, (_, i) => i);
    this.rank = new Array(n).fill(0);
  }

  // O(⍺(n))
  find(x) {
    if (this.parent[x] !== x) {
      this.parent[x] = this.find(this.parent[x]);
    }

    return this.parent[x];
  }

  // O(⍺(n))
  union(x, y) {
    const rootX = this.find(x);
    const rootY = this.find(y);

    if (rootX === rootY) return false;

    if (this.rank[rootX] < this.rank[rootY]) {
      this.parent[rootX] = rootY;
    } else if (this.rank[rootX] > this.rank[rootY]) {
      this.parent[rootY] = rootX;
    } else {
      this.parent[rootY] = rootX;
      this.rank[rootX] += 1;
    }

    return true;
  }
}

// Complexity:
// - Time: O(E ⍺(V))
// - Space: O(V)
class Solution {
  validTree(n, edges) {
    if (edges.length !== n - 1) return false;

    const ds = new DisjointSet(n);

    for (const [u, v] of edges) {
      if (!ds.union(u, v)) {
        return false;
      }
    }

    return true;
  }
}
```
