## [695 Max Area of Island](https://leetcode.com/problems/max-area-of-island/description/)

<!-- notecardId: 1755202222484 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/iJGr1OtmH0c

// Complexity:
// - Time: O(rows * cols)
// - Space: O(rows * cols)
function maxAreaOfIsland(grid) {
  const rows = grid.length;
  const cols = grid[0].length;
  let result = 0;

  for (let r = 0; r < rows; r += 1) {
    for (let c = 0; c < cols; c += 1) {
      if (grid[r][c] === 1) {
        result = Math.max(result, dfs(r, c));
      }
    }
  }

  return result;

  function dfs(row, col) {
    if (
      row < 0 ||
      col < 0 ||
      row >= rows ||
      col >= cols ||
      grid[row][col] === 0
    ) {
      return 0;
    }

    grid[row][col] = 0;
    let area = 1;

    area += dfs(row + 1, col);
    area += dfs(row - 1, col);
    area += dfs(row, col + 1);
    area += dfs(row, col - 1);

    return area;
  }
}
```
