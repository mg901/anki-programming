## [1436 Destination City](https://leetcode.com/problems/destination-city/description/)

<!-- notecardId: 1759099209953 -->

```js
// - Time: O(n)
// - Space: O(n)
function destCity(paths) {
  const outgoing = new Set();

  for (const [from] of paths) {
    outgoing.add(from);
  }

  for (const [, to] of paths) {
    if (!outgoing.has(to)) {
      return to;
    }
  }
}
```
