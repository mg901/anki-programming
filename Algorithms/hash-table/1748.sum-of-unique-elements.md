## [1748 Sum of Unique Elements](https://leetcode.com/problems/sum-of-unique-elements/description/)

<!-- notecardId: 1758845132263 -->

```js
// - Time: O(n)
// - Space: O(n)
function sumOfUnique(nums) {
  const map = new Map();
  let sum = 0;

  for (const num of nums) {
    if (!map.has(num)) {
      map.set(num, 1);
      sum += num;
    } else if (map.get(num) === 1) {
      sum -= num;
      map.set(num, 2);
    }
  }

  return sum;
}
```
