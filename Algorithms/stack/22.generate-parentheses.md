## [22 Generate Parentheses](https://leetcode.com/problems/generate-parentheses/description/)

<!-- notecardId: 1759245842310 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/s9fokUqJ76A

// - Time: O(4/^n / √n)
// - Space: O(4/^n / √n)
function generateParenthesis(n) {
  const result = [];
  const stack = [
    {
      current: '',
      open: 0,
      close: 0,
    },
  ];

  while (stack.length) {
    const { current, open, close } = stack.pop();

    if (current.length === 2 * n) {
      result.push(current);

      continue;
    }

    if (open < n) {
      stack.push({
        current: `${current}(`,
        open: open + 1,
        close,
      });
    }

    if (close < open) {
      stack.push({
        current: `${current})`,
        open,
        close: close + 1,
      });
    }
  }

  return result;
}
```
