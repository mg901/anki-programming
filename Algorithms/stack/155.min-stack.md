## [155 Min Stack](https://leetcode.com/problems/min-stack/description/)

<!-- notecardId: 1759185392819 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/qkLl7nAwDPo
class MinStack {
  #stack = [];

  push(val) {
    const min = !this.#stack.length
      ? val
      : Math.min(val, this.#stack.at(-1).min);

    this.#stack.push({
      val,
      min,
    });
  }

  pop() {
    this.#stack.pop();
  }

  top() {
    return this.#stack.at(-1).val;
  }

  getMin() {
    return this.#stack.at(-1).min;
  }
}
```
