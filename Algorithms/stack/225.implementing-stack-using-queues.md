## [225 Implement Stack using Queues 'two-queues'](https://leetcode.com/problems/implement-stack-using-queues/description/)

<!-- notecardId: 1764088743296 -->

```js
class MyStack {
  q1 = new Queue();

  q2 = new Queue();

  // - Time: O(n)
  push(x) {
    this.q2.enqueue(x);

    while (!this.q1.isEmpty()) {
      this.q2.enqueue(this.q1.dequeue());
    }

    const temp = this.q1;
    this.q1 = this.q2;
    this.q2 = temp;
  }

  pop() {
    return this.q1.dequeue();
  }

  top() {
    return this.q1.front();
  }

  empty() {
    return this.q1.isEmpty();
  }
}
```

## [225 Implement Stack using Queues 'one-queue'](https://leetcode.com/problems/implement-stack-using-queues/description/)

```js
class MyStack {
  queue = new Queue();

  // - Time: O(n)
  push(x) {
    this.queue.enqueue(x);
    const rotations = this.queue.size() - 1;

    for (let i = rotations; i > 0; i -= 1) {
      this.queue.enqueue(this.queue.dequeue());
    }
  }

  pop() {
    return this.queue.dequeue();
  }

  top() {
    return this.queue.front();
  }

  empty() {
    return this.queue.isEmpty();
  }
}
```
