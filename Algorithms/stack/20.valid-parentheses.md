## [20 Valid Parentheses](https://leetcode.com/problems/valid-parentheses/description/)

<!-- notecardId: 1743968394893 -->

```js
function isValid(s) {
  if (s.length < 2) return false;

  const stack = [];
  const map = new Map([
    [")", "("],
    ["]", "["],
    ["}", "{"],
  ]);

  for (const char of s) {
    if (!map.has(char)) {
      stack.push(char);
    } else if (map.get(char) !== stack.pop()) {
      return false;
    }
  }

  return stack.length === 0;
}
```
