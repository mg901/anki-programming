## [20 Valid Parentheses](https://leetcode.com/problems/valid-parentheses/description/)

<!-- notecardId: 1741174896978 -->

```js
function isValid(str) {
  const stack = [];
  const map = new Map([
    [")", "("],
    ["}", "{"],
    ["]", "["],
  ]);

  for (const char of str) {
    if (!map.has(char)) {
      stack.push(char);
    } else if (map.get(char) !== stack.pop()) return false;
  }

  return !stack.length;
}
```
