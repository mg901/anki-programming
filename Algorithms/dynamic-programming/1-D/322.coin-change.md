## [322 Coin Change](https://leetcode.com/problems/coin-change/description/)

```js
// Sub-pattern:
// - Unbounded Knapsack

// Top-down
// - Time: O(n * amount) or (n^amount) without memo
// - Space: O(amount)
function coinChange(coins, amount) {
  const cache = new Int32Array(amount + 1).fill(-1);
  const result = dfs(0);

  return result > amount ? -1 : result;

  function dfs(x) {
    if (x === amount) return 0;
    if (x > amount) return amount + 1; // Instead of Infinity
    if (cache[x] !== -1) return cache[x];

    let min = amount + 1;

    for (const coin of coins) {
      min = Math.min(min, 1 + dfs(x + coin));
    }

    cache[x] = min;

    return min;
  }
}

// Bottom-up
// - Time:
// - Space: O(amount)
function coinChange(coins, amount) {
  const dp = new Array(amount + 1).fill(Infinity);
  dp[0] = 0;

  for (let i = 1; i <= amount; i += 1) {
    for (const coin of coins) {
      if (i >= coin) {
        dp[i] = Math.min(dp[i], 1 + dp[i - coin]);
      }
    }
  }

  return dp[amount] === Infinity ? -1 : dp[amount];
}
```
