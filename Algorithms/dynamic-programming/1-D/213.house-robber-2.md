// Explanation:
// - Neetcode: https://youtu.be/rWAJCfYYOvM

// Sub-pattern:
// - Fibonacci numbers

// Bottom-up
// - Time: O(n)
// - Space: O(n)
function rob(nums) {
  const n = nums.length;

  if (n === 0) return 0;
  if (n === 1) return nums[0];

  return Math.max(toRob(nums.slice(0, -1)), toRob(nums.slice(1)));
}

// - Time: O(n)
// - Space: O(n)
function toRob(nums) {
  const n = nums.length;

  if (n === 0) return 0;
  if (n === 1) return nums[0];

  const [first, second] = nums;

  const dp = new Uint32Array(n);
  dp[0] = first;
  dp[1] = Math.max(first, second);

  for (let i = 2; i < n; i += 1) {
    dp[i] = Math.max(nums[i] + dp[i - 2], dp[i - 1]);
  }

  return dp[n - 1];
}
```

## [213 House Robber II 'space-optimized'](https://leetcode.com/problems/house-robber-ii/description/)

// Space Optimized
// - Time: O(n)
// - Space: O(1)
function rob(nums) {
  const n = nums.length;

  if (n === 0) return 0;
  if (n === 1) return nums[0];

  return Math.max(toRob(nums.slice(0, -1)), toRob(nums.slice(1)));
}

// - Time: O(n)
// - Space: O(1)
function rob(nums) {
  let twoBack = 0;
  let oneBack = 0;

  for (const num of nums) {
    const current = Math.max(num + twoBack, oneBack);
    twoBack = oneBack;
    oneBack = current;
  }

  return oneBack;
}
```
