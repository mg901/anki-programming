## [213 House Robber II](https://leetcode.com/problems/house-robber-ii/description/)

// Sub-pattern:
// - Fibonacci numbers

// Bottom-up
// - Time: O(n)
// - Space: O(n)
function rob(nums) {
  const n = nums.length;

  if (n === 0) return 0;
  if (n === 1) return nums[0];

  return Math.max(toRob(nums.slice(0, -1)), toRob(nums.slice(1)));
}

// - Time: O(n)
// - Space: O(n)
function toRob(nums) {
  const n = nums.length;

  if (n === 0) return 0;
  if (n === 1) return nums[0];

  const [first, second] = nums;

  const dp = new Uint32Array(n);
  dp[0] = first;
  dp[1] = Math.max(first, second);

  for (let i = 2; i < n; i += 1) {
    dp[i] = Math.max(nums[i] + dp[i - 2], dp[i - 1]);
  }

  return dp[n - 1];
}

// Space Optimized
// - Time: O(n)
// - Space: O(1)
function rob(nums) {
  const n = nums.length;

  if (n === 0) return 0;
  if (n === 1) return nums[0];

  return Math.max(toRob(nums.slice(0, -1)), toRob(nums.slice(1)));
}

// - Time: O(n)
// - Space: O(1)
function toRob(nums) {
  let prev = 0;
  let curr = 0;

  for (const num of nums) {
    const temp = Math.max(curr, prev + num);
    prev = curr;
    curr = temp;
  }

  return curr;
}
