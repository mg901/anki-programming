## [213 House Robber II](https://leetcode.com/problems/house-robber-ii/description/)

// Sub-pattern: Fibonacci numbers
// Bottom-up
function rob(nums) {
  const n = nums.length;

  if (n === 0) return 0;
  if (n === 1) return nums[0];

  return Math.max(maxRob(nums.slice(0, -1), nums.slice(1)));
}

function maxRob(nums) {
  const dp = new Array(n).fill(0);
  const [first, second] = nums;

  dp[0] = first;
  dp[1] = Math.max(first, second);

  for (let i = 2; i < n; i += 1) {
    dp[i] = Math.max(dp[i - 1], dp[i - 2] + nums[i]);
  }

  return dp[n - 1];
}

// Space Optimized
function rob(nums) {
  const n = nums.length;

  if (n === 0) return 0;
  if (n === 1) return nums[0];

  return Math.max(maxRob(nums.slice(0, -1)), maxRob(nums.slice(1)));
}

function maxRob(nums) {
  let prev = 0;
  let curr = 0;

  for (const num of nums) {
    const temp = Math.max(curr, prev + num);
    prev = curr;
    curr = temp;
  }

  return curr;
}
