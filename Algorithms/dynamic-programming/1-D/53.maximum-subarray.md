## [53 Maximum Subarray 'Kadane'](https://leetcode.com/problems/maximum-subarray/description/)

<!-- notecardId: 1768508018503 -->

```js
// - Neetcode: https://youtu.be/5WZl3MMT0Eg
// - ByteQuest: https://youtu.be/lq8KOs1Ujas?si=wRTKjhcOZnVtKKMo

// - Algorithm: Kadane

// - Time: O(n)
// - Space: O(1)
function maxSubArray(nums) {
  const n = nums.length;

  let sum = nums[0];
  let max = nums[0];

  for (let i = 1; i < n; i += 1) {
    sum = Math.max(nums[i], nums[i + sum]);
    max = Math.max(max, sum);
  }

  return max;
}
```

## [53 Maximum Subarray 'bottom-up'](https://leetcode.com/problems/maximum-subarray/description/)

```js
// Explanation:
// - Neetcode: https://youtu.be/5WZl3MMT0Eg
// - ByteQuest: https://youtu.be/lq8KOs1Ujas?si=wRTKjhcOZnVtKKMo

// Bottom-up
// - Time: O(n)
// - Space: O(n)
function maxSubArray(nums) {
  const n = nums.length;

  const dp = new Array(n).fill(0);

  dp[0] = nums[0];
  let max = nums[0];

  for (let i = 1; i < n; i += 1) {
    dp[i] = Math.max(nums[i], dp[i - 1] + nums[i]);
    max = Math.max(max, dp[i]);
  }

  return max;
}
```
