## [70 Climbing Stairs](https://leetcode.com/problems/climbing-stairs/description/)

```js
// Sub-pattern:
// - Fibonacci numbers

// Top-down
// - Time: O(n) or O(2^n) without memo
// - Space: O(n)
function climbStairs(n, cache = new Uint32Array(n + 1)) {
  if (cache[n] !== 0) return cache[n];
  if (n <= 1) return 1;

  cache[n] = climbStairs(n - 1, cache) + climbStairs(n - 2, cache);

  return cache[n];
}

// Bottom-up
// - Time: O(n)
// - Space: O(n)
function climbStairs(n) {
  const dp = new Uint32Array(n + 1).fill(1);

  for (let i = 2; i <= n; i += 1) {
    dp[i] = dp[i - 1] + dp[i - 2];
  }

  return dp[n];
}

// Space Optimized
// - Time: O(n)
// - Space: O(1)
function climbStairs(n) {
  let prev = 1; // fib(0)
  let current = 1; // fib(1)

  // fib(2 ...n - 1)
  for (let i = 1; i < n; i += 1) {
    let temp = current;
    current += prev;

    prev = temp;
  }

  return current;
}
```
