## [70 Climbing Stairs](https://leetcode.com/problems/climbing-stairs/description/)

<!-- notecardId: 1755875092051 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/Y0lT9Fck7qI

// Sub-pattern:
// - Fibonacci numbers

// Top-down
//
// Without memo:
// - Time:  O(2^n)
// - Space: O(n)
//
// With memo:
// - Time: O(n)
// - Space: O(n)
function climbStairs(n, memo = new Uint32Array(n + 1)) {
  if (memo[n] !== 0) return memo[n];
  if (n <= 1) return 1;

  memo[n] = climbStairs(n - 1, memo) + climbStairs(n - 2, memo);

  return memo[n];
}

// Bottom-up
// - Time: O(n)
// - Space: O(n)
function climbStairs(n) {
  const dp = new Uint32Array(n + 1).fill(1);

  for (let i = 2; i <= n; i += 1) {
    dp[i] = dp[i - 1] + dp[i - 2];
  }

  return dp[n];
}

// Space Optimized
// - Time: O(n)
// - Space: O(1)
function climbStairs(n) {
  let oneBack = 1;
  let twoBack = 1;

  for (let i = 1; i < n; i += 1) {
    let current = oneBack + twoBack;
    twoBack = oneBack;
    oneBack = current;
  }

  return oneBack;
}
```
