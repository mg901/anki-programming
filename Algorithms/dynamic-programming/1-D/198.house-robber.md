// Without memo:
// - Time:  O(2^n)
// - Space: O(n)
//
// With memo:
// - Time: O(n)
// - Space: O(n)
function rob(nums) {
  const n = nums.length;
  if (n === 0) return 0;
  if (n === 1) return nums[0];

  const memo = new Map();

  return dfs(0);

  function dfs(i) {
    if (memo.has(i)) return memo.get(i);
    if (i >= nums.length) return 0;

    let max = Math.max(nums[i] + dfs(i + 2), dfs(i + 1));
    memo.set(i, max);

    return max;
  }
}

## [198 House Robber 'bottom-up'](https://leetcode.com/problems/house-robber/description/)

```js
// Explanation:
// - Neetcode: https://youtu.be/73r3KWiEvyk

// Sub-pattern:
// - Fibonacci numbers

// Bottom-up
// - Time: O(n)
// - Space: O(n)
function rob(nums) {
  const n = nums.length;

  if (n === 0) return 0;
  if (n === 1) return nums[0];

  const [first, second] = nums;

  const dp = new Uint32Array(n);
  dp[0] = first;
  dp[1] = Math.max(first, second);

  for (let i = 2; i < n; i += 1) {
    dp[i] = Math.max(nums[i] + dp[i - 2], dp[i - 1]);
  }

  return dp[n - 1];
}
```

## [198 House Robber 'space-optimized'](https://leetcode.com/problems/house-robber/description/)

<!-- notecardId: 1756303359363 -->

```js
// Space Optimized
// - Time: O(n)
// - Space: O(1)
function rob(nums) {
  let twoBack = 0;
  let oneBack = 0;

  for (const num of nums) {
    const current = Math.max(num + twoBack, oneBack);
    twoBack = oneBack;
    oneBack = current;
  }

  return oneBack;
}
```
