## [198 House Robber](https://leetcode.com/problems/house-robber/description/)

// Sub-pattern:
// - Fibonacci numbers

// Top-down
// - Time: O(2^n)
// - Space: O(n)
function rob(nums) {
  const n = nums.length;
  if (n === 0) return 0;
  if (n === 1) return nums[0];
  if (n === 2) return Math.max(nums[0], nums[1]);

  return dfs(0);

  function dfs(index) {
    if (index >= nums.length) return 0;

    return Math.max(nums[index] + dfs(index + 2), dfs(index + 1));
  }
}

// Bottom-up
// - Time: O(n)
// - Space: O(n)
function rob(nums) {
  const n = nums.length;

  if (n === 0) return 0;
  if (n === 1) return nums[0];

  const [first, second] = nums;

  const dp = new Uint32Array(n);
  dp[0] = first;
  dp[1] = Math.max(first, second);

  for (let i = 2; i < n; i += 1) {
    dp[i] = Math.max(nums[i] + dp[i - 2], dp[i - 1]);
  }

  return dp[n - 1];
}

// Space Optimized
// - Time: O(n)
// - Space: O(1)
function rob(nums) {
  let twoBehind = 0;
  let oneBehind = 0;

  for (const num of nums) {
    const temp = Math.max(twoBehind + num, oneBehind);
    twoBehind = oneBehind;
    oneBehind = temp;
  }

  return oneBehind;
}
