## [198 House Robber](https://leetcode.com/problems/house-robber/description/)

// Explanation:
// - Neetcode: https://youtu.be/73r3KWiEvyk

// Sub-pattern:
// - Fibonacci numbers

// Top-down
//
// Without memo:
// - Time:  O(2^n)
// - Space: O(n)
//
// With memo:
// - Time: O(n)
// - Space: O(n)
function rob(nums) {
  const n = nums.length;
  const memo = new Int32Array(n).fill(-1);

  if (n === 0) return 0;
  if (n === 1) return nums[0];
  if (n === 2) return Math.max(nums[0], nums[1]);

  return dfs(0);

  function dfs(i) {
    if (i >= nums.length) return 0;
    if (memo[i] !== -1) return memo[i];

    memo[i] = Math.max(nums[i] + dfs(i + 2), dfs(i + 1));

    return memo[i];
  }
}

// Bottom-up
// - Time: O(n)
// - Space: O(n)
function rob(nums) {
  const n = nums.length;

  if (n === 0) return 0;
  if (n === 1) return nums[0];

  const [first, second] = nums;

  const dp = new Uint32Array(n);
  dp[0] = first;
  dp[1] = Math.max(first, second);

  for (let i = 2; i < n; i += 1) {
    dp[i] = Math.max(nums[i] + dp[i - 2], dp[i - 1]);
  }

  return dp[n - 1];
}

// Space Optimized
// - Time: O(n)
// - Space: O(1)
function rob(nums) {
  let twoBehind = 0;
  let oneBehind = 0;

  for (const num of nums) {
    const temp = Math.max(twoBehind + num, oneBehind);
    twoBehind = oneBehind;
    oneBehind = temp;
  }

  return oneBehind;
}
