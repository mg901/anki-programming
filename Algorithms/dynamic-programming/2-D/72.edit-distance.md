## [72 Edit Distance](https://leetcode.com/problems/edit-distance/description/)

// Explanation:
// - Neetcode: https://youtu.be/XYi2-LPrwm4
// - Back To Back SWE: https://youtu.be/MiqoA-yF-0M?si=zuZml_0ModEYia-X

// Top-down
//
// Without memo:
// - Time: O(3^(min(m, n)))
//   where:
//    m = `i`
//    n = `j`
// - Space: O(m + n)
//
// With memo:
// - Time: O(m * n)
// - Space: O(m * n)
function minDistance(word1, word2) {
  const memo = new Map();

  return dfs(0, 0);

  function dfs(i, j) {
    const key = `${i},${j}`;
    if (memo.has(key)) return memo.get(key);

    if (i === word1.length) return word2.length - j; // insert remaining characters
    if (j === word2.length) return word1.length - i; // delete remaining characters

    let result;

    if (word1[i] === word2[j]) {
      result = dfs(i + 1, j + 1);
    }

    result =
      1 +
      Math.min(
        dfs(i + 1, j), // delete
        dfs(i, j + 1), // insert
        dfs(i + 1, j + 1), // replace
      );

    memo.set(key, result);

    return result;
  }
}

// Bottom-up
// - Time: O(m * n)
// - Space: O(m * n)
function minDistance(word1, word2) {
  const m = word1.length;
  const n = word2.length;

  const dp = Array.from({ length: m + 1 }, () => new Uint32Array(n + 1));

  for (let i = 0; i <= m; i += 1) dp[i][0] = i;
  for (let j = 0; j <= n; j += 1) dp[0][j] = j;

  for (let i = 1; i <= m; i += 1) {
    for (let j = 1; j <= n; j += 1) {
      if (word1[i - 1] === word2[j - 1]) {
        dp[i][j] = dp[i - 1][j - 1];
      } else {
        dp[i][j] =
          1 +
          Math.min(
            dp[i - 1][j], // delete
            dp[i][j - 1], // insert
            dp[i - 1][j - 1], // replace
          );
      }
    }
  }

  return dp[m][n];
}

// Space Optimized
// - Time: O(m * n)
// - Space: O(n)
function minDistance(word1, word2) {
  const m = word1.length;
  const n = word2.length;

  if (n > m) return minDistance(word2, word1); // ensure word2 is shorter

  let prev = new Uint32Array(n + 1);
  let curr = new Uint32Array(n + 1);

  for (let j = 0; j <= n; j += 1) prev[j] = j;

  for (let i = 1; i <= m; i += 1) {
    curr[0] = i;

    for (let j = 1; j <= n; j += 1) {
      if (word1[i - 1] === word2[j - 1]) {
        curr[j] = prev[j - 1];
      } else {
        curr[j] =
          1 +
          Math.min(
            prev[j], // delete
            curr[j - 1], // insert
            prev[j - 1], // replace
          );
      }
    }

    [prev, curr] = [curr, prev]; // swap rows
  }

  return prev[n];
}
