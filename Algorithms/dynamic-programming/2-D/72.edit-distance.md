## [72 Edit Distance 'top-down'](https://leetcode.com/problems/edit-distance/description/)

<!-- notecardId: 1758393186987 -->

```js
// Top-down
//
// Without memo:
// - Time: O(3^(min(m, n)))
//   where:
//    m = `i`
//    n = `j`
// - Space: O(m + n)
//
// With memo:
// - Time: O(m * n)
// - Space: O(m * n)
function minDistance(word1, word2) {
  const memo = new Map();
  const m = word1.length;
  const n = word2.length;

  if (m > n) {
    [word1, word2] = [word2, word1];
  }

  return dfs(0, 0);

  function dfs(i, j) {
    const key = `${i},${j}`;
    if (memo.has(key)) return memo.get(key);

    if (i === m) return n - j; // insert remaining characters
    if (j === n) return m - i; // delete remaining characters

    let result;

    if (word1[i] === word2[j]) {
      result = dfs(i + 1, j + 1);
    }

    result =
      1 +
      Math.min(
        dfs(i + 1, j + 1), // replace
        dfs(i + 1, j), // delete
        dfs(i, j + 1), // insert
      );

    memo.set(key, result);

    return result;
  }
}
```

## [72 Edit Distance 'bottom-up'](https://leetcode.com/problems/edit-distance/description/)

// Explanation:
// - Neetcode: https://youtu.be/XYi2-LPrwm4

// - Time: O(m * n)
// - Space: O(m * n)
function minDistance(word1, word2) {
  const m = word1.length;
  const n = word2.length;

  if (m > n) {
    [word1, word2] = [word2, word1];
  }

  const dp = Array.from({ length: m + 1 }, () => new Uint32Array(n + 1));

  for (let i = 0; i <= m; i += 1) dp[i][0] = i;
  for (let j = 0; j <= n; j += 1) dp[0][j] = j;

  for (let i = 1; i <= m; i += 1) {
    for (let j = 1; j <= n; j += 1) {
      if (word1[i - 1] === word2[j - 1]) {
        dp[i][j] = dp[i - 1][j - 1];
      } else {
        dp[i][j] =
          1 +
          Math.min(
            dp[i][j - 1], // Insert
            dp[i - 1][j], // Delete
            dp[i - 1][j - 1], // Replace
          );
      }
    }
  }

  return dp[m][n];
}

// Space Optimized
// - Time: O(m * n)
// - Space: O(n)
function minDistance(word1, word2) {
  const m = word1.length;
  const n = word2.length;

  if (m > n) {
    [word1, word2] = [word2, word1];
  }

  let prev = new Uint32Array(n + 1);
  let curr = new Uint32Array(n + 1);

  for (let j = 0; j <= n; j += 1) prev[j] = j;

  for (let i = 1; i <= m; i += 1) {
    curr[0] = i;

    for (let j = 1; j <= n; j += 1) {
      if (word1[i - 1] === word2[j - 1]) {
        curr[j] = prev[j - 1];
      } else {
        curr[j] =
          1 +
          Math.min(
            prev[j - 1], // replace
            prev[j], // delete
            curr[j - 1], // insert
          );
      }
    }

    [prev, curr] = [curr, prev]; // swap rows
  }

  return prev[n];
}
