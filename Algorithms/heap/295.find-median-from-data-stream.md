## [295 Find Median from Data Stream](https://leetcode.com/problems/find-median-from-data-stream/description/)

<!-- notecardId: 1767295897424 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/itmhHWaHupI
class MedianFinder {
  #left = new MaxHeap();

  #right = new MinHeap();

  // Time: O(log (n))
  addNum(num) {
    if (this.#right.isEmpty() || num > this.#right.top()) {
      this.#right.push(num);
    } else {
      this.#left.push(num);
    }

    this.#rebalance();
  }

  // Time: O(1)
  findMedian() {
    if (this.#right.size() > this.#left.size()) {
      return this.#right.top();
    }

    return (this.#left.top() + this.#right.top()) / 2;
  }

  #rebalance() {
    if (this.#right.size() > this.#left.size() + 1) {
      this.#left.push(this.#right.pop());
    } else if (this.#left.size() > this.#right.size()) {
      this.#right.push(this.#left.pop());
    }
  }
}
```
