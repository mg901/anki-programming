## [567 Permutation in String](https://leetcode.com/problems/permutation-in-string/description/)

// Explanation:
// - Neetcode: https://youtu.be/gqXU1UyA8pk

// - Time: O(n)
// - Space: O(1)
function checkInclusion(s1, s2) {
  const n = s1.length;
  const m = s2.length;

  const ALPHABET_SIZE = 26;
  const codeA = 'a'.codePointAt(0);
  const s1Freq = new Uint8Array(ALPHABET_SIZE);
  const s2Freq = new Uint8Array(ALPHABET_SIZE);
  let matches = 0;

  for (let i = 0; i < n; i += 1) {
    s1Freq[s1.codePointAt(i) - codeA] += 1;
    s2Freq[s2.codePointAt(i) - codeA] += 1;
  }

  for (let i = 0; i < ALPHABET_SIZE; i += 1) {
    if (s1Freq[i] === s2Freq[i]) matches += 1;
  }

  for (let i = n; i < m; i += 1) {
    if (matches === ALPHABET_SIZE) return true;

    const rightIndex = s2.codePointAt(i) - codeA;
    const leftIndex = s2.codePointAt(i - n) - codeA;

    if (s1Freq[rightIndex] === s2Freq[rightIndex]) matches -= 1;
    s2Freq[rightIndex] += 1;
    if (s1Freq[rightIndex] === s2Freq[rightIndex]) matches += 1;

    if (s1Freq[leftIndex] === s2Freq[leftIndex]) matches -= 1;
    s2Freq[leftIndex] -= 1;
    if (s1Freq[leftIndex] === s2Freq[leftIndex]) matches += 1;
  }

  return matches === ALPHABET_SIZE;
}
