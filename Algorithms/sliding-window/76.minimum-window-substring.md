## [76 Minimum Window Substring](https://leetcode.com/problems/minimum-window-substring/description/)

<!-- notecardId: 1765549884925 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/jSto0O4AJbM

// Sub-pattern:
// - Variable-size window

// - Time: O(n + m)
//    where:
//     n - s.length
//     m - t.length
//
// - Space: O(k)
//    where:
//      k - unique characters of t
function minWindow(s, t) {
  const m = s.length;
  const n = t.length;

  if (n > m) return '';

  let need = n;
  const counter = new Map();
  let start = 0;
  let minSize = Infinity;

  for (const char of t) {
    counter.set(char, (counter.get(char) ?? 0) + 1);
  }

  let left = 0;

  for (let right = 0; right < m; right += 1) {
    const rightC = s[right];

    if (counter.has(rightC)) {
      if (counter.get(rightC) > 0) need -= 1;
      counter.set(rightC, counter.get(rightC) - 1);
    }

    while (need === 0) {
      const windowSize = right - left + 1;

      if (windowSize < minSize) {
        start = left;
        minSize = windowSize;
      }

      const leftC = s[left];

      if (counter.has(leftC)) {
        counter.set(leftC, counter.get(leftC) + 1);
        if (counter.get(leftC) > 0) need += 1;
      }

      left += 1;
    }
  }

  return minSize === Infinity ? '' : s.slice(start, start + minSize);
}
```
