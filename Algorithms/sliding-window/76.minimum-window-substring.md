## [76 Minimum Window Substring](https://leetcode.com/problems/minimum-window-substring/description/)

<!-- notecardId: 1759949694377 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/jSto0O4AJbM

// Sub-pattern:
// - Variable-size window

// - Time: O(n + m)
//    where:
//     n - s.length
//     m - t.length
//
// - Space: O(k)
//    where:
//      k - unique characters of t
function minWindow(s, t) {
  if (t.length === 0) return '';

  const charFreq = new Map();
  let need = t.length;
  let start = 0;
  let min = Infinity;

  for (const char of t) {
    charFreq.set(char, (charFreq.get(char) ?? 0) + 1);
  }

  let left = 0;
  for (let right = 0; right < s.length; right += 1) {
    const rc = s[right];

    if (charFreq.has(rc)) {
      if (charFreq.get(rc) > 0) need -= 1;
      charFreq.set(rc, charFreq.get(rc) - 1);
    }

    while (need === 0) {
      const windowSize = right - left + 1;

      if (windowSize < min) {
        start = left;
        min = windowSize;
      }

      const lc = s[left];

      if (charFreq.has(lc)) {
        charFreq.set(lc, charFreq.get(lc) + 1);
        if (charFreq.get(lc) > 0) need += 1;
      }

      left += 1;
    }
  }

  return min === Infinity ? '' : s.slice(start, start + min);
}
```
