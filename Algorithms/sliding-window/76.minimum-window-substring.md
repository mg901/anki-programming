## [76 Minimum Window Substring](https://leetcode.com/problems/minimum-window-substring/description/)

<!-- notecardId: 1742653056871 -->

```js
function minWindow(s, t) {
  if (s.length < t.length) return "";

  const charFreqMap = new Map();

  for (const char of t) {
    charFreqMap.set(char, (charFreqMap.get(char) ?? 0) + 1);
  }

  let toCover = t.length;
  let minLength = Infinity;
  let windowStart = 0;
  let left = 0;

  for (let right = 0; right < s.length; right += 1) {
    const char = s[right];

    if (charFreqMap.has(char)) {
      if (charFreqMap.get(char) > 0) toCover -= 1;
      charFreqMap.set(char, charFreqMap.get(char) - 1);
    }

    while (toCover === 0) {
      const windowLength = right - left + 1;

      if (windowLength < minLength) {
        minLength = windowLength;
        windowStart = left;
      }

      const leftChar = s[left];

      if (charFreqMap.has(leftChar)) {
        charFreqMap.set(leftChar, charFreqMap.get(leftChar) + 1);
        if (charFreqMap.get(leftChar) > 0) toCover += 1;
      }

      left += 1;
    }
  }

  return minLength === Infinity
    ? ""
    : s.slice(windowStart, windowStart + minLength);
}
```
