## [239 Sliding Window Maximum](https://leetcode.com/problems/sliding-window-maximum/description/)

<!-- notecardId: 1764333275562 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/DfljaUwZsOk

// Sub-pattern:
// - Variable-size window

// Technique:
// - Monotonic Decreasing Deque

// - Time: O(n)
// - Space: O(n)
function maxSlidingWindow(nums, k) {
  const n = nums.length;
  const deque = new Array(k);
  const result = [];
  let left = 0;
  let right = 0;

  for (let i = 0; i < n; i += 1) {
    if (i - k + 1 > deque[left]) {
      left += 1;
    }

    while (left < right && nums[i] >= nums[deque[right - 1]]) {
      right -= 1;
    }

    deque[right] = i;
    right += 1;

    if (i >= k - 1) {
      result.push(nums[deque[left]]);
    }
  }

  return result;
}
```
