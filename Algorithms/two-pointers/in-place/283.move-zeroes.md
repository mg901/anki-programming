## [Move Zeroes](https://leetcode.com/problems/move-zeroes/description/)

<!-- notecardId: 1739891677616 -->

```js
function moveZeroes(nums) {
  const n = nums.length;
  let right = 0;
  let left = 0;

  while (right < n) {
    if (nums[right] !== 0) {
      swap(nums, left, right);
      left += 1;
    }

    right += 1;
  }
}

function swap(arr, from, to) {
  const temp = arr[to];
  arr[to] = arr[from];
  arr[from] = temp;
}
```
