## [15 3Sum](https://leetcode.com/problems/3sum/description/)

<!-- notecardId: 1759074291886 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/jzZsG8n2R9A

// - Time: O(n^2)
// - Space: O(1)
function threeSum(nums) {
  const n = nums.length;
  if (n < 3) return [];

  nums.sort((a, b) => a - b);

  const result = [];

  for (let left = 0; left < n - 2; left += 1) {
    if (left > 0 && nums[left] === nums[left - 1]) continue;
    if (nums[left] > 0) break;

    let mid = left + 1;
    let right = n - 1;

    while (mid < right) {
      const sum = nums[left] + nums[mid] + nums[right];

      if (sum < 0) {
        mid += 1;
      } else if (sum > 0) {
        right -= 1;
      } else {
        result.push([nums[left], nums[mid], nums[right]]);

        while (nums[mid] === nums[mid + 1]) mid += 1;
        while (nums[right] === nums[right - 1]) right -= 1;

        mid += 1;
        right -= 1;
      }
    }
  }

  return result;
}
```
