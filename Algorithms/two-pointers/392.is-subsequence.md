## [392 Is Subsequence](https://leetcode.com/problems/is-subsequence/description)

<!-- notecardId: 1763931061625 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/99RVfqklbCE

// - Time: O(n + m)
// - Space: O(1)
function isSubsequence(s, t) {
  const m = s.length;
  const n = t.length;

  let i = 0;
  let j = 0;

  while (i < m && j < n) {
    if (s[i] === t[j]) {
      i += 1;
    }
    j += 1;
  }

  return i === m;
}
```
