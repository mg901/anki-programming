## [392 Is Subsequence](https://leetcode.com/problems/is-subsequence/description)

<!-- notecardId: 1765668041322 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/99RVfqklbCE

// - Time: O(n + m)
// - Space: O(1)
function isSubsequence(s, t) {
  const n = s.length;
  const m = t.length;

  let i = 0;
  let j = 0;

  while (i < n && j < m) {
    if (s[i] === t[j]) {
      i += 1;
    }

    j += 1;
  }

  return i === n;
}
```
