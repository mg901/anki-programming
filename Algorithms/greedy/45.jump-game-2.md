## [45 Jump Game 2](https://leetcode.com/problems/jump-game/description/)

<!-- notecardId: 1746888951353 -->

```js
function jump(nums) {
  let left = 0;
  let right = 0;
  let result = 0;

  while (right < nums.length - 1) {
    let farthest = 0;

    for (let i = left; i <= right; i += 1) {
      farthest = Math.max(farthest, i + nums[i]);
    }

    left = right + 1;
    right = farthest;
    result += 1;
  }

  return result;
}
```
