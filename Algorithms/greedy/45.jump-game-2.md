## [45 Jump Game 2](https://leetcode.com/problems/jump-game-ii/description/)

<!-- notecardId: 1768512841692 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/dJ7sWiOoK7g

// - Time: O(n)
// - Space: O(1)
function jump(nums) {
  const n = nums.length;

  let end = 0;
  let farthest = 0;
  let jumps = 0;

  for (let i = 0; i < n - 1; i += 1) {
    farthest = Math.max(farthest, i + nums[i]);

    if (i === end) {
      jumps += 1;
      end = farthest;
    }
  }

  return jumps;
}
```
