## [134 Gas Station](https://leetcode.com/problems/gas-station/description/)

<!-- notecardId: 1768651229059 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/lJwbPZGo05A

// - Time: O(n)
// - Space: O(1)
function canCompleteCircuit(gas, cost) {
  const n = gas.length;

  let totalGas = 0;
  let totalCost = 0;
  let start = 0;
  let tank = 0;

  for (let i = 0; i < n; i += 1) {
    totalGas += gas[i];
    totalCost += cost[i];
    tank += gas[i] - cost[i];

    if (tank < 0) {
      start = i + 1;
      tank = 0;
    }
  }

  return totalCost <= totalGas ? start : -1;
}
```
