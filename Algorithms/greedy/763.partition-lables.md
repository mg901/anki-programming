## [763 Partition Labels](https://leetcode.com/problems/partition-labels/description/)

```js
// Explanation:
// - Neetcode:

// - Time: O(n)
// - Space: O(1)
function partitionLabels(s) {
  const ALPHABET_SIZE = 26;
  const lastIndexes = new Uint8Array(ALPHABET_SIZE);
  const codePointA = 'a'.codePointAt(0);
  const n = s.length;

  for (let i = 0; i < n; i += 1) {
    lastIndexes[s.codePointAt(i) - codePointA] = i;
  }

  const result = [];
  let size = 0;
  let end = 0;

  for (let i = 0; i < n; i += 1) {
    size += 1;
    end = Math.max(end, lastIndexes[s.codePointAt(i) - codePointA]);

    if (i === end) {
      result.push(size);
      size = 0;
    }
  }

  return result;
}
```
