## [1899 Merge Triplets to Form Target Triplet](https://leetcode.com/problems/merge-triplets-to-form-target-triplet/description/)

<!-- notecardId: 1768660187264 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/kShkQLQZ9K4

// - Time: O(n)
// - Space: O(1)
function mergeTriplets(triplets, target) {
  let hasA = false;
  let hasB = false;
  let hasC = false;

  for (const [a, b, c] of triplets) {
    if (a > target[0] || b > target[1] || c > target[2]) {
      continue;
    }

    if (a === target[0]) hasA = true;
    if (b === target[1]) hasB = true;
    if (c === target[2]) hasC = true;
  }

  return hasA && hasB && hasC;
}
```
