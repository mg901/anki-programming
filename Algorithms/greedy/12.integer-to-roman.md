## [12 Integer to Roman](https://leetcode.com/problems/integer-to-roman/description/)

<!-- notecardId: 1768493684740 -->

```js
const romanNumerals = [
  { symbol: 'M', value: 1000 },
  { symbol: 'CM', value: 900 },
  { symbol: 'D', value: 500 },
  { symbol: 'CD', value: 400 },
  { symbol: 'C', value: 100 },
  { symbol: 'XC', value: 90 },
  { symbol: 'L', value: 50 },
  { symbol: 'XL', value: 40 },
  { symbol: 'X', value: 10 },
  { symbol: 'IX', value: 9 },
  { symbol: 'V', value: 5 },
  { symbol: 'IV', value: 4 },
  { symbol: 'I', value: 1 },
];

//  - Time: O(n)
// - Space: O(1)
function intToRoman(int) {
  let result = '';

  for (const { symbol, value } of romanNumerals) {
    while (int >= value) {
      result += symbol;
      int -= value;
    }
  }

  return result;
}
```
