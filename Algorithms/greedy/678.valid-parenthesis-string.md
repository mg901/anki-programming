## [678 Valid Parenthesis String](https://leetcode.com/problems/valid-parenthesis-string/description/)

<!-- notecardId: 1768742036102 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/QhPdNS143Qg

// - Time: O(n)
// - Space: O(1)
function checkValidString(str) {
  let openMin = 0;
  let openMax = 0;

  for (const char of str) {
    if (char === '(') {
      openMin += 1;
      openMax += 1;
    } else if (char === ')') {
      openMin -= 1;
      openMax -= 1;
    } else {
      openMax += 1;
      openMin -= 1;
    }

    if (openMax < 0) return false;
    if (openMin < 0) openMin = 0;
  }

  return openMin === 0;
}
```
