## [678 Valid Parenthesis String](https://leetcode.com/problems/valid-parenthesis-string/description/)

<!-- notecardId: 1769516562174 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/QhPdNS143Qg

// - Time: O(n)
// - Space: O(1)
function checkValidString(str) {
  let openMin = 0;
  let openMax = 0;

  for (const char of str) {
    if (char === "(") {
      openMin += 1;
      openMax += 1;
    } else if (char === ")") {
      openMin -= 1;
      openMax -= 1;
    } else {
      // char === '*'
      openMax += 1; // treat '*' as '('
      openMin -= 1; // treat '*' as ')'
    }

    if (openMax < 0) return false; // Too many closing parentheses
    if (openMin < 0) openMin = 0; // Can't have negative minimum
  }

  return openMin === 0; // All open parentheses are closed
}
```
