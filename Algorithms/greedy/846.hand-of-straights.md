## [846 Hand of Straights](https://leetcode.com/problems/hand-of-straights/description/)

<!-- notecardId: 1768657006613 -->

```js
// Explanation:
// - Neetcode: https://youtu.be/amnrMCVd2YI

// - Time: O(n log(n))
// - Space: O(n)
function isNStraightHand(hand, groupSize) {
  const n = hand.length;

  if (n % groupSize !== 0) return false;

  hand = hand.toSorted((a, b) => a - b);
  const counter = new Map();

  for (const num of hand) {
    counter.set(num, (counter.get(num) ?? 0) + 1);
  }

  for (const num of hand) {
    if (counter.get(num) === 0) continue;

    for (let i = 0; i < groupSize; i += 1) {
      const next = num + i;

      if ((counter.get(next) ?? 0) === 0) {
        return false;
      }

      counter.set(next, counter.get(next) - 1);
    }
  }

  return true;
}
```
